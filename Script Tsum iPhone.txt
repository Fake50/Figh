local function decodeItemData(jsonString)
    if not jsonString or type(jsonString) ~= "string" then return nil end
    if #jsonString == 0 then return nil end
    local first, last = string.sub(jsonString, 1, 1), string.sub(jsonString, -1)
    if first == "}" and last == "{" then
        jsonString = string.reverse(jsonString)
    end
    local success, data = pcall(function()
        return HttpService:JSONDecode(jsonString)
    end)
    if success and data then
        return data
    else
        return nil
    end
end

-- Функция поиска по ID предмета из списка
local function findItemById(assetId)
    return ALL_ITEMS[tostring(assetId)]
end

-- Проверка, является ли объект частью игрока
local function isPlayerModel(obj)
    -- Поднимаемся по иерархии и проверяем, не находится ли объект в игроке
    local current = obj
    while current and current ~= workspace do
        -- Проверяем, есть ли Humanoid (признак персонажа)
        if current:FindFirstChildOfClass("Humanoid") then
            -- Проверяем, принадлежит ли персонаж игроку
            for _, player in pairs(Players:GetPlayers()) do
                if player.Character == current then
                    return true
                end
            end
        end
        current = current.Parent
    end
    return false
end

local function findAllItems()
    local results = {}
    local processedModels = {} -- Чтобы не добавлять дубликаты
    
    -- Ищем по всем объектам в workspace
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("Pants") or obj:IsA("Shirt") or obj:IsA("Accessory") or obj:IsA("MeshPart") then
            -- ВАЖНО: Пропускаем объекты, которые находятся в игроках
            if isPlayerModel(obj) then
                continue
            end
            
            local assetId = nil
            local itemType = "Item"
            
            -- Получаем AssetId из Pants/Shirt
            if obj:IsA("Pants") then
                assetId = string.match(tostring(obj.PantsTemplate), "%d+")
                itemType = "Clothing"
            elseif obj:IsA("Shirt") then
                assetId = string.match(tostring(obj.ShirtTemplate), "%d+")
                itemType = "Clothing"
            elseif obj:IsA("Accessory") then
                -- Ищем Handle в аксессуаре
                local handle = obj:FindFirstChild("Handle")
                if handle then
                    -- Проверяем SpecialMesh
                    local specialMesh = handle:FindFirstChildOfClass("SpecialMesh")
                    if specialMesh and specialMesh.MeshId then
                        assetId = string.match(tostring(specialMesh.MeshId), "%d+")
                    end
                    
                    -- Если не нашли, проверяем MeshPart
                    if not assetId then
                        local meshPart = handle:FindFirstChildOfClass("MeshPart")
                        if meshPart and meshPart.MeshId then
                            assetId = string.match(tostring(meshPart.MeshId), "%d+")
                        end
                    end
                    
                    -- Если не нашли, проверяем TextureID
                    if not assetId and specialMesh and specialMesh.TextureId then
                        assetId = string.match(tostring(specialMesh.TextureId), "%d+")
                    end
                end
                itemType = "Accessory"
            elseif obj:IsA("MeshPart") and obj.MeshId ~= "" then
                -- ЛОГИКА ИЗ ESP ACS: Ищем аксессуары в моделях M_*
                -- Проверяем, что это Handle, Body, L или R
                if obj.Name == "Handle" or obj.Name == "Body" or obj.Name == "L" or obj.Name == "R" then
                    local parentModel = obj.Parent
                    -- Проверяем что родитель - это модель M_*
                    if parentModel and parentModel:IsA("Model") and parentModel.Name:match("^M_") then
                        assetId = string.match(tostring(obj.MeshId), "%d+")
                        itemType = "Accessory"
                    end
                end
            end
            
            if assetId then
                local itemData = findItemById(assetId)
                if itemData then
                    -- Определяем целевой объект
                    local target = obj
                    
                    if obj:IsA("Accessory") then
                        -- Для аксессуаров проверяем, есть ли родительская Model
                        if obj.Parent:IsA("Model") and obj.Parent ~= workspace then
                            target = obj.Parent
                        else
                            target = obj  -- Сам аксессуар как цель
                        end
                    elseif obj:IsA("MeshPart") then
                        -- Для MeshPart (Handle, Body, L, R) используем родительскую Model M_*
                        if obj.Parent and obj.Parent:IsA("Model") then
                            target = obj.Parent
                        else
                            target = obj
                        end
                    else
                        -- Для одежды поднимаемся до Model
                        target = obj.Parent
                        while target and not target:IsA("Model") do
                            target = target.Parent
                            if target == workspace or target == game then
                                target = nil
                                break
                            end
                        end
                    end
                    
                    -- Проверяем, что это не игрок и не дубликат
                    if target and not isPlayerModel(target) and not processedModels[target] then
                        processedModels[target] = true
                        
                        table.insert(results, {
                            object = target,
                            name = itemData.name,
                            price = 0,
                            resellMulti = 0,
                            rarity = itemData.rarity,
                            type = itemType,
                            quality = "",
                            assetId = assetId,
                        })
                    end
                end
            end
        end
    end
    return results
end

local function getDistance(obj)
    local char = player.Character
    if not char then return nil end
    local root = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso")
    if not root then return nil end
    local pos
    if obj:IsA("BasePart") then
        pos = obj.Position
    elseif obj:IsA("Model") then
        if obj.PrimaryPart then
            pos = obj.PrimaryPart.Position
        else
            local part = obj:FindFirstChildWhichIsA("BasePart")
            if part then pos = part.Position end
        end
    end
    if not pos then return nil end
    return (root.Position - pos).Magnitude
end

local function createESP(obj, item)
    if not obj or not obj.Parent then return end
    if obj:FindFirstChild("ESP_Billboard") then return end

    -- Находим самую высокую видимую часть предмета
    local targetPart = nil
    local highestY = -math.huge
    
    if obj:IsA("Model") then
        -- Проходим по всем частям модели и находим самую высокую
        for _, part in pairs(obj:GetDescendants()) do
            if part:IsA("BasePart") and part.Transparency < 1 then
                local partY = part.Position.Y + (part.Size.Y / 2)
                if partY > highestY then
                    highestY = partY
                    targetPart = part
                end
            end
        end
        
        -- Если не нашли прозрачные части, берем PrimaryPart или первую BasePart
        if not targetPart then
            targetPart = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
        end
    elseif obj:IsA("BasePart") then
        targetPart = obj
    end
    
    if not targetPart then return end

    local distance = getDistance(obj)
    local distText = distance and string.format("%.0f м", distance) or ""

    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP_Billboard"
    billboard.Adornee = targetPart  -- Прикрепляем к самой высокой части
    billboard.Size = UDim2.new(0, 300, 0, 70)
    billboard.StudsOffset = Vector3.new(0, 3, 0)  -- Поднимаем выше
    billboard.AlwaysOnTop = true
    billboard.Parent = targetPart  -- Родитель - сама часть, а не модель

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 1, 0)
    frame.BackgroundTransparency = 1
    frame.Parent = billboard

    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(1, 0, 0.5, 0)
    nameLabel.Position = UDim2.new(0, 0, 0, 0)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = item.name
    nameLabel.TextColor3 = rarityColors[item.rarity] or Color3.fromRGB(255, 255, 255)
    nameLabel.Font = Enum.Font.SourceSansBold
    nameLabel.TextSize = 16
    nameLabel.TextStrokeTransparency = 0.3
    nameLabel.Parent = frame

    local infoLabel = Instance.new("TextLabel")
    infoLabel.Size = UDim2.new(1, 0, 0.5, 0)
    infoLabel.Position = UDim2.new(0, 0, 0.5, 0)
    infoLabel.BackgroundTransparency = 1
    infoLabel.Text = string.format("[%s]  %s", item.rarity, distText)
    infoLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    infoLabel.Font = Enum.Font.SourceSans
    infoLabel.TextSize = 14
    infoLabel.TextStrokeTransparency = 0.3
    infoLabel.Parent = frame
end

-- Плавный полет к объекту с TweenService
local function tweenToObject(targetObj, checkForNewItems)
    local char = player.Character
    if not char then return false, nil end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return false, nil end
    
    local targetPart = targetObj
    if targetObj:IsA("Model") then
        targetPart = targetObj.PrimaryPart or targetObj:FindFirstChildWhichIsA("BasePart")
    end
    
    if not targetPart or not targetPart:IsA("BasePart") then
        print("[AutoBuy] Не удалось найти позицию для полета")
        return false, nil
    end
    
    -- Вычисляем расстояние и время полета
    local distance = (root.Position - targetPart.Position).Magnitude
    local duration = distance / autoBuySpeed
    
    print(string.format("[AutoBuy] Лечу к цели (%.0f м, %.1f сек)", distance, duration))
    
    -- Создаем Tween для плавного перемещения
    local targetCFrame = CFrame.new(targetPart.Position) * CFrame.new(0, 3, 0)
    local tweenInfo = TweenInfo.new(duration, Enum.EasingStyle.Linear)
    local tween = TweenService:Create(root, tweenInfo, {CFrame = targetCFrame})
    
    tween:Play()
    tween.Completed:Wait()
    
    print("[AutoBuy] ✅ Прилетел к цели")
    return true, nil
end

-- Определение ShopZone для объекта (предмета или Seller)
local function findShopZone(obj)
    -- Получаем позицию объекта
    local objPos = nil
    if obj:IsA("Model") then
        if obj.PrimaryPart then
            objPos = obj.PrimaryPart.Position
        else
            local part = obj:FindFirstChildWhichIsA("BasePart")
            if part then objPos = part.Position end
        end
    elseif obj:IsA("BasePart") then
        objPos = obj.Position
    end
    
    if not objPos then return nil end
    
    -- Ищем все ShopZone в workspace
    for _, zone in pairs(workspace:GetDescendants()) do
        if zone.Name:match("^Shop_ShopZone") and zone:IsA("BasePart") then
            -- Проверяем находится ли объект внутри зоны
            local zonePos = zone.Position
            local zoneSize = zone.Size
            
            -- Проверка по осям X, Y, Z
            local inX = math.abs(objPos.X - zonePos.X) <= zoneSize.X / 2
            local inY = math.abs(objPos.Y - zonePos.Y) <= zoneSize.Y / 2
            local inZ = math.abs(objPos.Z - zonePos.Z) <= zoneSize.Z / 2
            
            if inX and inY and inZ then
                return zone
            end
        end
    end
    
    return nil
end

-- Поиск ближайшего Seller в той же ShopZone (с возможностью исключить конкретного)
local function findSellerInShop(itemObj, excludeSeller)
    print("[AutoBuy] 🔍 Ищу Seller в той же ShopZone...")
    
    -- Получаем текущую позицию игрока
    local char = player.Character
    if not char then 
        print("[AutoBuy] ❌ Персонаж не найден")
        return nil 
    end
    
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then 
        print("[AutoBuy] ❌ HumanoidRootPart не найден")
        return nil 
    end
    
    local playerPos = root.Position
    local playerY = playerPos.Y
    print(string.format("[AutoBuy] 📍 Позиция игрока: X=%.0f Y=%.0f Z=%.0f", playerPos.X, playerPos.Y, playerPos.Z))
    
    -- Определяем ShopZone предмета
    local itemZone = findShopZone(itemObj)
    if itemZone then
        print("[AutoBuy] 🏪 Предмет находится в зоне: " .. itemZone.Name)
    else
        print("[AutoBuy] ⚠️ Не удалось определить ShopZone предмета")
    end
    
    -- Глобальный поиск всех Seller в workspace
    local allSellers = {}
    
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj.Name == "Seller" and obj:IsA("Model") and obj:FindFirstChild("HumanoidRootPart") then
            -- Исключаем конкретного seller, если указан
            if not excludeSeller or obj ~= excludeSeller then
                table.insert(allSellers, obj)
            else
                print("[AutoBuy] ⚠️ Исключаю предыдущего Seller из списка")
            end
        end
    end
    
    if #allSellers == 0 then
        print("[AutoBuy] ❌ Не найдено ни одного Seller в игре!")
        return nil
    end
    
    print("[AutoBuy] ✅ Найдено Seller в игре: " .. #allSellers)
    
    -- Фильтруем Seller по той же ShopZone
    local sellersInZone = {}
    
    if itemZone then
        for _, seller in pairs(allSellers) do
            local sellerZone = findShopZone(seller)
            if sellerZone and sellerZone == itemZone then
                table.insert(sellersInZone, seller)
                print("[AutoBuy] ✅ Seller в той же зоне: " .. itemZone.Name)
            else
                if sellerZone then
                    print("[AutoBuy] ❌ Seller в другой зоне: " .. sellerZone.Name)
                else
                    print("[AutoBuy] ⚠️ Seller не в зоне")
                end
            end
        end
    end
    
    -- Если нашли Seller в той же зоне, ищем среди них
    local searchList = #sellersInZone > 0 and sellersInZone or allSellers
    
    if #sellersInZone > 0 then
        print("[AutoBuy] ✅ Найдено Seller в той же ShopZone: " .. #sellersInZone)
    else
        print("[AutoBuy] ⚠️ Seller в той же зоне не найден, ищу среди всех")
    end
    
    -- Выбираем ближайшего Seller (на том же этаже)
    local closestSeller = nil
    local closestDist = math.huge
    local maxHeightDiff = 15 -- Максимальная разница по высоте
    
    for _, seller in pairs(searchList) do
        local sellerRoot = seller:FindFirstChild("HumanoidRootPart")
        if sellerRoot then
            local sellerPos = sellerRoot.Position
            local sellerY = sellerPos.Y
            
            -- Проверяем разницу по высоте
            local heightDiff = math.abs(playerY - sellerY)
            
            -- Если Seller на том же этаже
            if heightDiff <= maxHeightDiff then
                -- Считаем горизонтальную дистанцию
                local horizontalDist = math.sqrt(
                    (playerPos.X - sellerPos.X)^2 + 
                    (playerPos.Z - sellerPos.Z)^2
                )
                
                print(string.format("[AutoBuy] ✅ Seller подходит! Дистанция: %.0f м, высота: %.0f", horizontalDist, heightDiff))
                
                if horizontalDist < closestDist then
                    closestDist = horizontalDist
                    closestSeller = seller
                end
            else
                print(string.format("[AutoBuy] ⚠️ Seller на другом этаже (разница: %.0f)", heightDiff))
            end
        end
    end
    
    if closestSeller then
        local zone = findShopZone(closestSeller)
        local zoneName = zone and zone.Name or "неизвестная зона"
        print(string.format("[AutoBuy] ✅✅✅ Выбран Seller в зоне '%s': %.0f м", zoneName, closestDist))
        return closestSeller
    end
    
    -- Если не нашли на том же этаже, берем просто ближайшего из зоны
    print("[AutoBuy] ⚠️ Не найден Seller на том же этаже, ищу любого в зоне...")
    closestSeller = nil
    closestDist = math.huge
    
    for _, seller in pairs(searchList) do
        local sellerRoot = seller:FindFirstChild("HumanoidRootPart")
        if sellerRoot then
            local dist = (playerPos - sellerRoot.Position).Magnitude
            if dist < closestDist then
                closestDist = dist
                closestSeller = seller
            end
        end
    end
    
    if closestSeller then
        print(string.format("[AutoBuy] ✅ Выбран ближайший Seller: %.0f м", closestDist))
        return closestSeller
    end
    
    print("[AutoBuy] ❌ Не удалось выбрать Seller")
    return nil
end

-- Основной цикл автобая
local autoBuyCounter = 0

local function autoBuyCycle()
    if not autoBuyEnabled then return end
    
    print("\n[AutoBuy] ═══════════════════════════════")
    print("[AutoBuy] 🔄 Цикл покупки #" .. (autoBuyCounter + 1))
    
    -- 1. Находим предметы с ESP
    local items = findAllItems()
    if #items == 0 then
        print("[AutoBuy] ⏸️ Нет предметов с ESP, жду...")
        return
    end
    
    print("[AutoBuy] 📊 Найдено предметов: " .. #items)
    
    -- 2. Выбираем ближайший предмет
    table.sort(items, function(a, b)
        local distA = getDistance(a.object) or math.huge
        local distB = getDistance(b.object) or math.huge
        return distA < distB
    end)
    
    local targetItem = items[1]
    local distance = getDistance(targetItem.object)
    print(string.format("[AutoBuy] 🎯 Цель: %s (%s) - %.0f м", targetItem.name, targetItem.rarity, distance or 0))
    
    -- 3. Летим к предмету
    print("[AutoBuy] ✈️ Лечу к предмету...")
    local success, _ = tweenToObject(targetItem.object, false)
    if not success then
        print("[AutoBuy] ❌ Не удалось долететь до предмета")
        return
    end
    
    task.wait(0.5)
    
    -- 4. Берем предмет (клик)
    print("[AutoBuy] 🛒 Беру предмет в корзину...")
    if not clickObject(targetItem.object) then
        print("[AutoBuy] ❌ Не удалось взять предмет")
        return
    end
    
    print("[AutoBuy] ✅ Предмет взят в корзину")
    task.wait(0.5)
    
    -- 5. Ищем Seller в том же магазине
    local excludedSeller = nil  -- Seller которого нужно исключить из поиска
    local maxRetries = 3  -- Максимум 3 попытки с разными seller
    local retryCount = 0
    
    while retryCount < maxRetries do
        print("[AutoBuy] 🔍 Ищу продавца (Seller)...")
        if excludedSeller then
            print("[AutoBuy] ⚠️ Исключаю неправильного Seller из поиска")
        end
        
        local seller = findSellerInShop(targetItem.object, excludedSeller)
        if not seller then
            print("[AutoBuy] ❌ Seller не найден в магазине")
            return
        end
        
        -- 6. Летим к Seller
        print("[AutoBuy] ✈️ Лечу к продавцу...")
        local successFly, _ = tweenToObject(seller, false)
        if not successFly then
            print("[AutoBuy] ❌ Не удалось долететь до Seller")
            return
        end
        
        task.wait(0.3)
        
        -- 7. Покупаем (клик по Seller)
        print("[AutoBuy] � Покупаю предмет...")
        if not clickObject(seller) then
            print("[AutoBuy] ❌ Не удалось кликнуть по Seller")
            return
        end
        
        -- 8. ПРОВЕРКА: Продался ли предмет?
        task.wait(1.5)  -- Увеличено время ожидания для обработки на ПК
        
        -- Обновляем ESP и проверяем, остался ли предмет
        local itemsAfter = findAllItems()
        local itemStillExists = false
        
        for _, item in pairs(itemsAfter) do
            if item.assetId == targetItem.assetId then
                -- Проверяем, что это тот же предмет (по позиции)
                local dist = getDistance(item.object)
                if dist and dist < 100 then  -- Если предмет рядом
                    itemStillExists = true
                    targetItem = item  -- Обновляем ссылку на предмет
                    break
                end
            end
        end
        
        if not itemStillExists then
            -- Предмет исчез - покупка успешна!
            autoBuyCounter = autoBuyCounter + 1
            print("[AutoBuy] ✅✅✅ ПОКУПКА ЗАВЕРШЕНА! ✅✅✅")
            print("[AutoBuy] 📈 Всего куплено: " .. autoBuyCounter)
            print("[AutoBuy] ═══════════════════════════════\n")
            break
        else
            -- Предмет всё ещё есть - корзина была пустая!
            print("[AutoBuy] ⚠️⚠️⚠️ КОРЗИНА ПУСТАЯ! Предмет не был куплен!")
            print("[AutoBuy] � Возвращаюсь к предмету и ищу другого Seller...")
            
            -- Исключаем этого seller из следующего поиска
            excludedSeller = seller
            retryCount = retryCount + 1
            
            if retryCount >= maxRetries then
                print("[AutoBuy] ❌ Превышено количество попыток, пропускаю предмет")
                return
            end
            
            -- Возвращаемся к предмету
            print("[AutoBuy] ✈️ Лечу обратно к предмету...")
            local successReturn, _ = tweenToObject(targetItem.object, false)
            if not successReturn then
                print("[AutoBuy] ❌ Не удалось вернуться к предмету")
                return
            end
            
            task.wait(0.5)
            
            -- Снова берем предмет
            print("[AutoBuy] � Беру предмет в корзину заново...")
            if not clickObject(targetItem.object) then
                print("[AutoBuy] ❌ Не удалось взять предмет повторно")
                return
            end
            
            print("[AutoBuy] ✅ Предмет снова в корзине, ищу другого Seller...")
            task.wait(0.5)
            
            -- Цикл продолжится и найдет другого seller
        end
    end
    
    -- 9. Обновляем ESP
    task.wait(0.5)
    if espEnabled then
        updateESP()
    end
end

-- Поток автобая
local autoBuyThread = nil
local function startAutoBuy()
    if autoBuyThread then return end
    
    -- Диагностика доступных функций
    print("[AutoBuy] 🔍 Проверка доступных функций:")
    print("  fireclickdetector:", fireclickdetector ~= nil)
    print("  fireproximityprompt:", fireproximityprompt ~= nil)
    print("  firetouchinterest:", firetouchinterest ~= nil)
    print("  mouse1click:", mouse1click ~= nil)
    print("  mouse1press:", mouse1press ~= nil)
    print("  mouse1release:", mouse1release ~= nil)
    
    print("[AutoBuy] 🚀 AutoBuy запущен")
    autoBuyThread = task.spawn(function()
        while autoBuyEnabled do
            local success, err = pcall(function()
                autoBuyCycle()
            end)
            
            if not success then
                print("[AutoBuy] ⚠️ Ошибка в цикле: " .. tostring(err))
            end
            
            -- Короткая задержка между циклами (1 секунда)
            task.wait(1)
        end
        autoBuyThread = nil
        print("[AutoBuy] ⏹️ AutoBuy остановлен")
    end)
end

local function stopAutoBuy()
    autoBuyEnabled = false
    if autoBuyThread then
        task.cancel(autoBuyThread)
        autoBuyThread = nil
    end
    
    -- Возвращаем камеру в нормальный режим
    local camera = workspace.CurrentCamera
    camera.CameraType = Enum.CameraType.Custom
    print("[AutoBuy] 🛑 AutoBuy остановлен, камера восстановлена")
end

local function pressE(barigaNPC)
    print("[AutoSell] 🖱️ Нажимаю E для открытия GUI...")
    
    -- Ищем ProximityPrompt
    local proximityPrompt = nil
    for _, descendant in pairs(barigaNPC:GetDescendants()) do
        if descendant:IsA("ProximityPrompt") then
            proximityPrompt = descendant
            print("[AutoSell] ✅ Найден ProximityPrompt")
            break
        end
    end
    
    if proximityPrompt then
        -- Пробуем активировать ProximityPrompt несколькими способами
        print("[AutoSell] Активирую ProximityPrompt...")
        
        -- Метод 1: fireproximityprompt
        if fireproximityprompt then
            local success1 = pcall(function()
                fireproximityprompt(proximityPrompt)
            end)
            if success1 then
                print("[AutoSell] ✅ fireproximityprompt() успешно")
                task.wait(0.3)
                return true
            end
            
            local success2 = pcall(function()
                fireproximityprompt(proximityPrompt, 0)
            end)
            if success2 then
                print("[AutoSell] ✅ fireproximityprompt(0) успешно")
                task.wait(0.3)
                return true
            end
        end
        
        -- Метод 2: InputHoldBegin/End
        local success3 = pcall(function()
            proximityPrompt:InputHoldBegin()
            task.wait(proximityPrompt.HoldDuration or 0.1)
            proximityPrompt:InputHoldEnd()
        end)
        if success3 then
            print("[AutoSell] ✅ InputHold успешно")
            task.wait(0.3)
            return true
        end
        
        -- Метод 3: Triggered event
        local success4 = pcall(function()
            proximityPrompt.Triggered:Fire(player)
        end)
        if success4 then
            print("[AutoSell] ✅ Triggered:Fire успешно")
            task.wait(0.3)
            return true
        end
    end
    
    -- Метод 4: Симуляция клавиши E
    print("[AutoSell] Пробую симуляцию клавиши E...")
    local success = pcall(function()
        VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.E, false, game)
        task.wait(0.1)
        VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.E, false, game)
    end)
    
    if success then
        print("[AutoSell] ✅ Клавиша E нажата")
        task.wait(0.3)
        return true
    end
    
    print("[AutoSell] ⚠️ Все методы нажатия E выполнены")
    return true -- Возвращаем true чтобы продолжить, GUI может открыться
end

            end
        end
        return false
    end
    
    -- Функция для клика по GUI элементу
    local function clickGuiElement(element)
        if not element then return false end
        
        print("[AutoSell] 🖱️ Кликаю по: " .. element.Name .. " (" .. element.ClassName .. ")")
        
        -- Метод 1: VirtualInputManager (самый надёжный для любых элементов)
        local success1 = pcall(function()
            local absPos = element.AbsolutePosition
            local absSize = element.AbsoluteSize

-- Основной цикл автоселла
local function autoSellCycle()
    if not autoSellEnabled then return end
    
    print("\n[AutoSell] ═══════════════════════════════")
    print("[AutoSell] 🔄 Цикл продажи #" .. (autoSellCounter + 1))
    
    -- 1. Находим BarigaNPC
    local barigaNPC = findBarigaNPC()
    if not barigaNPC then
        print("[AutoSell] ❌ BarigaNPC не найден, жду...")
        return
    end
    
    local successInv2 = pcall(function()
        ReplicatedStorage:WaitForChild("InventoryRemotes"):WaitForChild("RequestInventory"):FireServer()
    end)
    print("[AutoSell] RequestInventory #2:", successInv2)
    
    task.wait(0.5)
    
    -- 2. Получаем инвентарь Бариги
    print("[AutoSell] 🔄 Получаю инвентарь Бариги...")
    local barigaInventory = nil
    local attempts = 0
    local maxAttempts = 3
    
    -- Пробуем получить инвентарь несколько раз
    while attempts < maxAttempts and (not barigaInventory or type(barigaInventory) ~= "table") do
        attempts = attempts + 1
        print(string.format("[AutoSell] Попытка #%d из %d...", attempts, maxAttempts))
        
        local success, err = pcall(function()
            barigaInventory = ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("GetBarigaInventory"):InvokeServer()
        end)
        
        if not success then
            print("[AutoSell] ⚠️ Ошибка при получении инвентаря: " .. tostring(err))
            task.wait(0.5)
        elseif type(barigaInventory) == "table" then
            local count = 0
            for _ in pairs(barigaInventory) do count = count + 1 end
            if count > 0 then
                break  -- Инвентарь получен и не пустой
            else
                print("[AutoSell] ⚠️ Инвентарь пустой, пробую еще раз...")
                task.wait(0.5)
            end
        else
            print("[AutoSell] ⚠️ Инвентарь не таблица, пробую еще раз...")
            task.wait(0.5)
        end
    end
    
    if not barigaInventory or type(barigaInventory) ~= "table" then
        print("[AutoSell] ❌ Не удалось получить инвентарь после " .. maxAttempts .. " попыток")
        return false
    end
    
    print("[AutoSell] ✅ Инвентарь Бариги получен (тип: table)")
    
    -- 3. Собираем все ID предметов для продажи
    local itemIds = {}
    
    -- Отладка: показываем структуру инвентаря
    local itemCount = 0
    for itemId, itemData in pairs(barigaInventory) do
        itemCount = itemCount + 1
        print(string.format("[AutoSell] 📝 Предмет #%d: ID=%s, Data=%s", itemCount, tostring(itemId), tostring(itemData)))
        table.insert(itemIds, tostring(itemId)) -- Преобразуем в строку
    end
    
    if #itemIds == 0 then
        print("[AutoSell] ⚠️ Нет предметов для продажи (инвентарь пуст)")
        
        -- Закрываем GUI перед выходом
        pcall(function()
            ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("CloseBariga"):FireServer()
        end)
        
        return false
    end
    
    print(string.format("[AutoSell] 📊 Найдено предметов для продажи: %d", #itemIds))
    
    -- 4. Получаем оффер для всех предметов
    print("[AutoSell] 💰 Получаю оффер для предметов...")
    print("[AutoSell] Отправляю IDs:", table.concat(itemIds, ", "))
    
    local offer = nil
    local successOffer, offerErr = pcall(function()
        -- ВАЖНО: Используем unpack с вложенным массивом (как в примере)
        -- local args = {{"id1", "id2", ...}}
        -- InvokeServer(unpack(args)) -> InvokeServer({"id1", "id2", ...})
        local args = {itemIds}
        offer = ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("GetBarigaOffer"):InvokeServer(unpack(args))
    end)
    
    if not successOffer then
        print("[AutoSell] ❌ Ошибка при получении оффера: " .. tostring(offerErr))
        
        -- Закрываем GUI перед выходом
        pcall(function()
            ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("CloseBariga"):FireServer()
        end)
        
        return false
    end
    
    print("[AutoSell] ✅ Оффер получен (тип: " .. type(offer) .. ")")
    
    -- Если оффер - таблица, показываем детали
    if type(offer) == "table" then
        for k, v in pairs(offer) do
            print(string.format("[AutoSell] Оффер[%s] = %s", tostring(k), tostring(v)))
        end
    end
    
    task.wait(0.5)
    
    -- 5. Подтверждаем продажу
    print("[AutoSell] 💵 Подтверждаю продажу...")
    local successSell, sellErr = pcall(function()
        -- ВАЖНО: Используем unpack с вложенным массивом (как в примере)
        -- local args = {true}
        -- FireServer(unpack(args)) -> FireServer(true)
        local args = {true}
        ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("ConfirmBarigaSale"):FireServer(unpack(args))
    end)
    
    if not successSell then
        print("[AutoSell] ❌ Ошибка при подтверждении продажи: " .. tostring(sellErr))
        
        -- Закрываем GUI перед выходом
        pcall(function()
            ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("CloseBariga"):FireServer()
        end)
        
        return false
    end
    
    print("[AutoSell] ✅✅✅ ПРОДАЖА ЗАВЕРШЕНА! ✅✅✅")
    task.wait(0.5)
    
    -- 6. Закрываем GUI Бариги
    print("[AutoSell] 🚪 Закрываю GUI...")
    pcall(function()
        ReplicatedStorage:WaitForChild("BarigaRemotes"):WaitForChild("CloseBariga"):FireServer()
    end)
    
    autoSellCounter = autoSellCounter + 1
    print(string.format("[AutoSell] 📈 Всего продаж: %d", autoSellCounter))
    
    return true
end

        print("[AutoSell] ⚠️ Цикл продажи завершен с ошибками")

-- Поток автоселла
local autoSellThread = nil
local function startAutoSell()
    if autoSellThread then return end
    
    print("[AutoSell] 🚀 AutoSell запущен")
    autoSellThread = task.spawn(function()
        while autoSellEnabled do
            local success, err = pcall(function()
                autoSellCycle()
            end)
            
            if not success then
                print("[AutoSell] ⚠️ Ошибка в цикле: " .. tostring(err))
            end
            
            -- Задержка между циклами (5 секунд)
            task.wait(5)
        end
        autoSellThread = nil
        print("[AutoSell] ⏹️ AutoSell остановлен")
    end)
end

local function stopAutoSell()
    autoSellEnabled = false
    if autoSellThread then
        task.cancel(autoSellThread)
        autoSellThread = nil
    end
    
    -- Возвращаем камеру в нормальный режим
    local camera = workspace.CurrentCamera
    camera.CameraType = Enum.CameraType.Custom
    print("[AutoSell] 🛑 AutoSell остановлен, камера восстановлена")
end
