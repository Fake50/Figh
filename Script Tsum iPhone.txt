local function getAllMannequins()
    local results = {}
    for _, zone in pairs(workspace:GetChildren()) do
        if zone:IsA("Model") and zone.Name:match("Shop_ShopZone") then
            local slots = zone:FindFirstChild("ItemSlots")
            if slots then
                for _, slot in pairs(slots:GetChildren()) do
                    if slot.Name:match("Slot_") then
                        local mannequin = slot:FindFirstChild("Mannequin")
                        if mannequin and mannequin:IsA("Model") then
                            -- НОВАЯ ПРОВЕРКА: Манекен должен быть потомком workspace
                            if not mannequin:IsDescendantOf(workspace) then
                                continue
                            end
                            
                            local clothingData = {}
                            local hasValidClothing = false
                            
                            for _, child in pairs(mannequin:GetChildren()) do
                                if child:IsA("Shirt") then
                                    local id = child.ShirtTemplate:match("%d+")
                                    clothingData.shirtId = id
                                    -- Проверяем что это вещь из базы
                                    if id and ASSETS[id] then
                                        hasValidClothing = true
                                    end
                                elseif child:IsA("Pants") then
                                    local id = child.PantsTemplate:match("%d+")
                                    clothingData.pantsId = id
                                    -- Проверяем что это вещь из базы
                                    if id and ASSETS[id] then
                                        hasValidClothing = true
                                    end
                                end
                            end
                            
                            -- Добавляем только если нашли валидную одежду из базы
                            if hasValidClothing and (clothingData.shirtId or clothingData.pantsId) then
                                table.insert(results, {
                                    mannequin = mannequin,
                                    clothingData = clothingData,
                                    slot = slot
                                })
                            end
                        end
                    end
                end
            end
        end
    end
    return results
end

local function getAllAccessories()
    local results = {}
    for _, zone in pairs(workspace:GetChildren()) do
        if zone:IsA("Model") and zone.Name:match("Shop_ShopZone") then
            for _, child in pairs(zone:GetChildren()) do
                if child:IsA("Model") and child.Name:match("^M_") then
                    -- НОВАЯ ПРОВЕРКА: Модель должна быть в workspace
                    if not child:IsDescendantOf(workspace) then
                        continue
                    end
                    
                    local target = child:FindFirstChild("Handle") or child:FindFirstChild("Body")
                    if target and target:IsA("MeshPart") and target.MeshId then
                        local id = target.MeshId:match("%d+")
                        -- Проверяем что это аксессуар из базы
                        if id and ACCESSORIES[id] then
                            table.insert(results, {
                                model = child,
                                meshId = id
                            })
                        end
                    end
                end
            end
        end
    end
    return results
end

local function createHighlight(model)
    if model:FindFirstChild("ESP_H") then return end
    
    local h = Instance.new("Highlight")
    h.Name = "ESP_H"
    h.Adornee = model
    h.FillColor = RARITY_COLOR
    h.OutlineColor = RARITY_COLOR
    h.FillTransparency = 0.5
    h.OutlineTransparency = 0
    h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    h.Parent = model
    table.insert(espObjects, h)
end

local function createLabel(model, text)
    if model:FindFirstChild("ESP_L") then 
        local existingGui = model:FindFirstChild("ESP_L")
        if existingGui then
            return existingGui
        end
    end
    
    local gui = Instance.new("BillboardGui")
    gui.Name = "ESP_L"
    gui.Adornee = model
    gui.Size = UDim2.new(0, 200, 0, 40)
    gui.StudsOffset = Vector3.new(0, 3, 0)
    gui.AlwaysOnTop = true
    gui.Parent = player:WaitForChild("PlayerGui")
    
    local label = Instance.new("TextLabel")
    label.Name = "ESP_Text" -- Добавляем имя для легкого поиска
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    label.BackgroundTransparency = 0.3
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.Font = Enum.Font.SourceSansBold
    label.TextSize = 14
    label.Text = text
    label.Parent = gui
    
    table.insert(espObjects, gui)
    return gui
end

local function clearESP()
    for _, obj in pairs(espObjects) do
        if obj and obj.Parent then
            obj:Destroy()
        end
    end
    espObjects = {}
    espTracking = {}
end

-- Функция для принудительного удаления ESP
local function forceRemoveESP(model, reason)
    if not espTracking[model] then return end
    
    local espData = espTracking[model]
    local itemName = ""
    
    if espData.label and espData.label:IsA("BillboardGui") then
        local label = espData.label:FindFirstChild("ESP_Text")
        if label and label:IsA("TextLabel") then
            itemName = label.Text
        end
    end
    
    pcall(function()
        if espData.highlight and espData.highlight.Parent then
            espData.highlight:Destroy()
        end
    end)
    pcall(function()
        if espData.label and espData.label.Parent then
            espData.label:Destroy()
        end
    end)
    
    espTracking[model] = nil
    
    if itemName ~= "" then
        print("🗑️ ESP удален (" .. reason .. "):", itemName)
    end
end

local function updateESP()
    -- Создаем список для удаления (чтобы избежать изменения таблицы во время итерации)
    local toRemove = {}
    
    for model, espData in pairs(espTracking) do
        local shouldRemove = false
        local removeReason = ""
        
        -- Проверка 1: Модель существует в workspace?
        if not model or not model.Parent then
            shouldRemove = true
            removeReason = "модель удалена"
        end
        
        -- Проверка 2: Модель все еще потомок workspace?
        if not shouldRemove then
            local success = pcall(function()
                if not model:IsDescendantOf(workspace) then
                    shouldRemove = true
                    removeReason = "модель не в workspace"
                end
            end)
            if not success then
                shouldRemove = true
                removeReason = "ошибка проверки модели"
            end
        end
        
        -- Проверка 3: Слот существует? (для вещей)
        if not shouldRemove and espData.slot then
            if not espData.slot.Parent then
                shouldRemove = true
                removeReason = "слот удален"
            else
                local success = pcall(function()
                    if not espData.slot:IsDescendantOf(workspace) then
                        shouldRemove = true
                        removeReason = "слот не в workspace"
                    end
                end)
                if not success then
                    shouldRemove = true
                    removeReason = "ошибка проверки слота"
                end
                
                if not shouldRemove then
                    -- Проверяем что манекен все еще в слоте
                    local mannequinInSlot = espData.slot:FindFirstChild("Mannequin")
                    if not mannequinInSlot then
                        shouldRemove = true
                        removeReason = "манекен удален из слота"
                    elseif mannequinInSlot ~= model then
                        shouldRemove = true
                        removeReason = "другой манекен в слоте"
                    end
                end
            end
        end
        
        -- Проверка 4: Для манекенов - есть ли еще одежда из базы?
        if not shouldRemove and model:IsA("Model") and model.Name == "Mannequin" then
            local hasTargetClothing = false
            local success = pcall(function()
                for _, child in pairs(model:GetChildren()) do
                    if child:IsA("Shirt") then
                        local id = child.ShirtTemplate:match("%d+")
                        if id and ASSETS[id] then
                            hasTargetClothing = true
                            break
                        end
                    elseif child:IsA("Pants") then
                        local id = child.PantsTemplate:match("%d+")
                        if id and ASSETS[id] then
                            hasTargetClothing = true
                            break
                        end
                    end
                end
            end)
            
            if not success or not hasTargetClothing then
                shouldRemove = true
                removeReason = "одежда пропала с манекена"
            end
        end
        
        -- Проверка 5: Для аксессуаров - существует ли Handle/Body с нужным MeshId?
        if not shouldRemove and model:IsA("Model") and model.Name:match("^M_") then
            local success = pcall(function()
                local target = model:FindFirstChild("Handle") or model:FindFirstChild("Body")
                if not target or not target:IsA("MeshPart") then
                    shouldRemove = true
                    removeReason = "Handle/Body удален"
                else
                    local meshId = target.MeshId:match("%d+")
                    if not meshId or not ACCESSORIES[meshId] then
                        shouldRemove = true
                        removeReason = "MeshId изменен или удален"
                    end
                end
            end)
            
            if not success then
                shouldRemove = true
                removeReason = "ошибка проверки аксессуара"
            end
        end
        
        -- Проверка 6: ESP элементы существуют?
        if not shouldRemove then
            if espData.highlight and not pcall(function() return espData.highlight.Parent end) then
                shouldRemove = true
                removeReason = "highlight поврежден"
            end
            if espData.label and not pcall(function() return espData.label.Parent end) then
                shouldRemove = true
                removeReason = "label поврежден"
            end
        end
        
        if shouldRemove then
            table.insert(toRemove, {model = model, reason = removeReason})
        end
    end
    
    -- Удаляем все помеченные ESP
    for _, data in ipairs(toRemove) do
        forceRemoveESP(data.model, data.reason)
    end
end

local function enableESP()
    if espEnabled then return end
    espEnabled = true
    
    clearESP()
    
    local found = 0
    
    -- Вещи
    for _, data in pairs(getAllMannequins()) do
        local names = {}
        if data.clothingData.shirtId then
            local item = ASSETS[data.clothingData.shirtId]
            if item then
                table.insert(names, item.name)
            end
        end
        if data.clothingData.pantsId then
            local item = ASSETS[data.clothingData.pantsId]
            if item then
                table.insert(names, item.name)
            end
        end
        
        if #names > 0 then
            local highlight = createHighlight(data.mannequin)
            local label = createLabel(data.mannequin, table.concat(names, " + "))
            espTracking[data.mannequin] = {
                highlight = data.mannequin:FindFirstChild("ESP_H"),
                label = label,
                slot = data.slot, -- Сохраняем ссылку на слот
                itemType = "clothing"
            }
            found = found + 1
        end
    end
    
    -- Аксессуары
    for _, data in pairs(getAllAccessories()) do
        local item = ACCESSORIES[data.meshId]
        if item then
            local highlight = createHighlight(data.model)
            local label = createLabel(data.model, item.name)
            espTracking[data.model] = {
                highlight = data.model:FindFirstChild("ESP_H"),
                label = label,
                itemType = "accessory"
            }
            found = found + 1
        end
    end
    
    -- Запускаем автоматическое обновление ESP
    task.spawn(function()
        while espEnabled do
            updateESP()
            task.wait(0.15) -- Проверяем каждые 0.15 секунды (быстрее реагирует)
        end
    end)
    
    -- Добавляем слушателя на удаление объектов для мгновенной реакции
    for model, trackData in pairs(espTracking) do
        if model and model.Parent then
            -- Следим за удалением модели
            model.AncestryChanged:Connect(function(_, parent)
                if parent == nil or not model:IsDescendantOf(workspace) then
                    -- Модель удалена из Workspace - мгновенно обновляем
                    task.wait(0.05)
                    updateESP()
                end
            end)
            
            -- Следим за удалением одежды у манекенов
            if model:IsA("Model") and model.Name == "Mannequin" then
                -- Следим за существующей одеждой
                for _, child in pairs(model:GetChildren()) do
                    if child:IsA("Shirt") or child:IsA("Pants") then
                        local clothingId = nil
                        if child:IsA("Shirt") then
                            clothingId = child.ShirtTemplate:match("%d+")
                        else
                            clothingId = child.PantsTemplate:match("%d+")
                        end
                        
                        -- Только для одежды из нашей базы
                        if clothingId and ASSETS[clothingId] then
                            child.AncestryChanged:Connect(function(_, parent)
                                if parent == nil or not child:IsDescendantOf(model) then
                                    -- Одежда удалена - мгновенно обновляем
                                    task.wait(0.05)
                                    updateESP()
                                end
                            end)
                            
                            -- Следим за изменением Template (если вещь заменили)
                            child:GetPropertyChangedSignal("ShirtTemplate"):Connect(function()
                                task.wait(0.05)
                                updateESP()
                            end)
                            if child:IsA("Pants") then
                                child:GetPropertyChangedSignal("PantsTemplate"):Connect(function()
                                    task.wait(0.05)
                                    updateESP()
                                end)
                            end
                        end
                    end
                end
                
                -- Следим за добавлением новых детей (вдруг одежда появилась)
                model.ChildRemoved:Connect(function(child)
                    if child:IsA("Shirt") or child:IsA("Pants") then
                        task.wait(0.05)
                        updateESP()
                    end
                end)
            end
            
            -- Следим за аксессуарами
            if model:IsA("Model") and model.Name:match("^M_") then
                local target = model:FindFirstChild("Handle") or model:FindFirstChild("Body")
                if target then
                    target.AncestryChanged:Connect(function(_, parent)
                        if parent == nil or not target:IsDescendantOf(model) then
                            task.wait(0.05)
                            updateESP()
                        end
                    end)
                end
            end
            
            -- Следим за слотом (для вещей)
            if trackData.slot and trackData.slot.Parent then
                trackData.slot.AncestryChanged:Connect(function(_, parent)
                    if parent == nil or not trackData.slot:IsDescendantOf(workspace) then
                        -- Слот удален - мгновенно обновляем
                        task.wait(0.05)
                        updateESP()
                    end
                end)
                
                -- Следим за изменением детей слота (манекен может быть заменен)
                trackData.slot.ChildRemoved:Connect(function(child)
                    if child == model or child.Name == "Mannequin" then
                        task.wait(0.05)
                        updateESP()
                    end
                end)
            end
        end
    end
    
    print("✅ ESP включен! Найдено:", found)
end

local function disableESP()
    espEnabled = false
    clearESP()
    print("❌ ESP выключен")
end

-- Получить баланс игрока
local function getPlayerMoney()
    -- Пробуем разные варианты получения денег
    local money = 0
    
    -- Вариант 1: Leaderstats
    pcall(function()
        local leaderstats = player:FindFirstChild("leaderstats")
        if leaderstats then
            local cash = leaderstats:FindFirstChild("Cash") or leaderstats:FindFirstChild("Money") or leaderstats:FindFirstChild("Coins")
            if cash and cash.Value then
                money = cash.Value
            end
        end
    end)
    
    -- Вариант 2: PlayerGui
    if money == 0 then
        pcall(function()
            local playerGui = player:WaitForChild("PlayerGui", 1)
            for _, gui in pairs(playerGui:GetDescendants()) do
                if gui:IsA("TextLabel") and gui.Name:lower():find("cash") or gui.Name:lower():find("money") then
                    local text = gui.Text
                    local amount = text:match("%d+")
                    if amount then
                        money = tonumber(amount) or 0
                    end
                end
            end
        end)
    end
    
    return money
end

-- Получить цену предмета из слота
local function getItemPrice(slot)
    -- Пробуем найти цену в слоте
    local price = 0
    
    pcall(function()
        -- Вариант 1: Атрибут Price
        if slot:GetAttribute("Price") then
            price = slot:GetAttribute("Price")
        end
        
        -- Вариант 2: StringValue или NumberValue с именем Price
        if price == 0 then
            local priceValue = slot:FindFirstChild("Price") or slot:FindFirstChild("price")
            if priceValue and (priceValue:IsA("NumberValue") or priceValue:IsA("IntValue")) then
                price = priceValue.Value
            elseif priceValue and priceValue:IsA("StringValue") then
                local amount = priceValue.Value:match("%d+")
                if amount then
                    price = tonumber(amount)
                end
            end
        end
        
        -- Вариант 3: Ищем в Configuration
        if price == 0 then
            for _, child in pairs(slot:GetDescendants()) do
                if child.Name:lower():find("price") then
                    if child:IsA("NumberValue") or child:IsA("IntValue") then
                        price = child.Value
                    elseif child:IsA("StringValue") then
                        local amount = child.Value:match("%d+")
                        if amount then
                            price = tonumber(amount)
                        end
                    end
                end
            end
        end
        
        -- Вариант 4: Ищем TextLabel с ценой рядом
        if price == 0 then
            for _, desc in pairs(slot:GetDescendants()) do
                if desc:IsA("TextLabel") or desc:IsA("BillboardGui") then
                    local text = ""
                    if desc:IsA("TextLabel") then
                        text = desc.Text
                    else
                        local label = desc:FindFirstChildOfClass("TextLabel")
                        if label then
                            text = label.Text
                        end
                    end
                    
                    -- Ищем паттерн цены: $123, 123$, Cash: 123, и т.д.
                    local amount = text:match("%$%s*(%d+)") or text:match("(%d+)%s*%$") or text:match("Cash:%s*(%d+)")
                    if amount then
                        price = tonumber(amount)
                        break
                    end
                end
            end
        end
    end)
    
    return price
end

-- Получить SlotId из Slot
local function getSlotId(slot)
    -- Слот может иметь разные структуры, пробуем все варианты
    
    -- Вариант 1: SlotId хранится в атрибуте
    if slot:GetAttribute("SlotId") then
        return slot:GetAttribute("SlotId")
    end
    
    -- Вариант 2: SlotId в StringValue или подобном
    local slotIdValue = slot:FindFirstChild("SlotId") or slot:FindFirstChild("slotId") or slot:FindFirstChild("ID")
    if slotIdValue and slotIdValue:IsA("StringValue") then
        return slotIdValue.Value
    end
    
    -- Вариант 3: Имя слота содержит полный SlotId (например "Slot_5587_114582")
    if slot.Name:match("Slot_%d+_%d+") then
        return slot.Name
    end
    
    -- Вариант 4: Имя слота это только "Slot_X", ищем Configuration или другие данные
    for _, child in pairs(slot:GetChildren()) do
        if child:IsA("Configuration") or child.Name == "Data" or child.Name == "Info" then
            for _, subChild in pairs(child:GetChildren()) do
                if subChild.Name:lower():find("slot") and subChild:IsA("StringValue") then
                    return subChild.Value
                end
            end
        end
    end
    
    -- Вариант 5: Проверяем все StringValue в слоте
    for _, child in pairs(slot:GetDescendants()) do
        if child:IsA("StringValue") and child.Value:match("SLOT_%d+_%d+") then
            return child.Value
        end
    end
    
    -- Если ничего не нашли, выводим структуру слота для отладки
    print("⚠️ Не удалось найти SlotId. Структура слота:")
    print("  Имя:", slot.Name)
    print("  Дети:")
    for _, child in pairs(slot:GetChildren()) do
        print("    -", child.Name, child.ClassName)
        if child:IsA("StringValue") or child:IsA("IntValue") or child:IsA("NumberValue") then
            print("      Значение:", child.Value)
        end
    end
    print("  Атрибуты:")
    for attr, value in pairs(slot:GetAttributes()) do
        print("    -", attr, "=", value)
    end
    
    -- В крайнем случае возвращаем имя слота
    return slot.Name
end

-- БЕЗОПАСНЫЙ ТЕЛЕПОРТ С ОБХОДОМ АНТИЧИТА
-- ============================================
-- Используем TweenService для МАКСИМАЛЬНО плавного перемещения
-- Античит видит это как обычное движение персонажа

local activeTween = nil -- Храним активный твин для отмены

-- УЛУЧШЕННЫЙ ОБХОД: Телепорт с имитацией человеческого движения
local function advancedSafeTeleport(startCF, endCF, distance)
    local character = player.Character
    if not character then return false end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    -- Отменяем предыдущий твин если есть
    if activeTween then
        activeTween:Cancel()
        activeTween = nil
    end
    
    -- ИСПРАВЛЕНИЕ: Сохраняем текущую ориентацию персонажа
    local currentLookVector = startCF.LookVector
    
    -- Вычисляем безопасную скорость (оптимизированная)
    local maxSpeed = 25 -- Скорость для обхода детекта (чуть быстрее)
    local calculatedDuration = distance / maxSpeed
    
    -- Минимум для небольших расстояний
    local safeDuration = math.max(calculatedDuration, 1.0)
    
    -- Для очень больших расстояний добавляем немного времени
    if distance > 120 then
        safeDuration = math.max(safeDuration, 2.0)
    end
    
    print("🛡️ Безопасный телепорт:", math.floor(distance), "стадов за", string.format("%.1f", safeDuration), "сек")
    print("⚡ Скорость:", string.format("%.1f", distance/safeDuration), "стадов/сек")
    
    -- Разбиваем на несколько сегментов для имитации человеческого движения
    local segments = math.max(2, math.floor(distance / 30)) -- Каждые 30 стадов = новый сегмент
    local segmentDuration = safeDuration / segments
    
    for i = 1, segments do
        local startProgress = (i - 1) / segments
        local endProgress = i / segments
        
        -- Добавляем небольшое случайное отклонение для имитации человека
        local randomOffset = Vector3.new(
            math.random(-100, 100) / 100, -- ±1 стад по X
            0,
            math.random(-100, 100) / 100  -- ±1 стад по Z
        )
        
        local segmentStart = startCF.Position:Lerp(endCF.Position, startProgress)
        local segmentEnd = startCF.Position:Lerp(endCF.Position, endProgress)
        
        -- Только последний сегмент идет точно в цель, остальные с отклонением
        if i < segments then
            segmentEnd = segmentEnd + randomOffset
        end
        
        -- ИСПРАВЛЕНИЕ: Создаем CFrame с сохранением направления взгляда
        local segmentStartCF = CFrame.new(segmentStart, segmentStart + currentLookVector)
        local segmentEndCF = CFrame.new(segmentEnd, segmentEnd + currentLookVector)
        
        -- Создаем твин для сегмента
        local tweenInfo = TweenInfo.new(
            segmentDuration,
            Enum.EasingStyle.Linear,
            Enum.EasingDirection.InOut,
            0,
            false,
            0
        )
        
        local tween = TweenService:Create(hrp, tweenInfo, {CFrame = segmentEndCF})
        activeTween = tween
        tween:Play()
        tween.Completed:Wait()
        
        -- Случайная микропауза между сегментами (имитация человека)
        if i < segments then
            task.wait(math.random(5, 15) / 100) -- 0.05-0.15 сек пауза
        end
    end
    
    -- ИСПРАВЛЕНИЕ: Финальная точная установка позиции с сохранением ориентации
    hrp.CFrame = CFrame.new(endCF.Position, endCF.Position + currentLookVector)
    activeTween = nil
    
    return true
end

local function smoothTweenTeleport(startCF, endCF, duration, easingStyle)
    local character = player.Character
    if not character then return false end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    -- Отменяем предыдущий твин если есть
    if activeTween then
        activeTween:Cancel()
        activeTween = nil
    end
    
    -- ИСПРАВЛЕНИЕ: Сохраняем текущую ориентацию персонажа (только Y поворот)
    local currentLookVector = startCF.LookVector
    local targetPosition = endCF.Position
    -- Создаем новый CFrame с сохранением направления взгляда
    local correctedEndCF = CFrame.new(targetPosition, targetPosition + currentLookVector)
    
    -- Создаем TweenInfo
    local tweenInfo = TweenInfo.new(
        duration,                    -- Длительность
        Enum.EasingStyle.Linear,     -- Linear = постоянная скорость без разгона
        Enum.EasingDirection.InOut,  -- Направление
        0,                           -- Повторы (0 = без повторов)
        false,                       -- Реверс
        0                            -- Задержка
    )
    
    -- Создаем твин для CFrame с сохраненной ориентацией
    local tween = TweenService:Create(hrp, tweenInfo, {CFrame = correctedEndCF})
    activeTween = tween
    
    -- Запускаем анимацию
    tween:Play()
    
    -- Ждем завершения
    tween.Completed:Wait()
    activeTween = nil
    
    return true
end

local function smoothTeleport(startPos, endPos, steps, delayBetweenSteps)
    local character = player.Character
    if not character then return false end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    -- ИСПРАВЛЕНИЕ: Сохраняем текущую ориентацию
    local currentLookVector = hrp.CFrame.LookVector
    
    -- Вычисляем расстояние и направление
    local distance = (endPos - startPos).Magnitude
    
    -- Если расстояние маленькое, просто телепортируемся
    if distance < 10 then
        hrp.CFrame = CFrame.new(endPos, endPos + currentLookVector)
        return true
    end
    
    -- Плавное перемещение маленькими шагами
    for i = 1, steps do
        local alpha = i / steps
        local nextPos = startPos:Lerp(endPos, alpha)
        
        -- Безопасная установка позиции с сохранением ориентации
        pcall(function()
            hrp.CFrame = CFrame.new(nextPos, nextPos + currentLookVector)
        end)
        
        task.wait(delayBetweenSteps)
    end
    
    -- Финальная установка точной позиции с сохранением ориентации
    pcall(function()
        hrp.CFrame = CFrame.new(endPos, endPos + currentLookVector)
    end)
    
    return true
end

-- Телепорт к позиции (с ПРОДВИНУТЫМ обходом античита)
local function teleportTo(position)
    local character = player.Character
    if not character then return false end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    
    local currentPos = hrp.Position
    local currentCF = hrp.CFrame
    local currentLookVector = currentCF.LookVector -- Сохраняем направление взгляда
    local targetCF = CFrame.new(position, position + currentLookVector) -- Создаем целевой CFrame с сохранением ориентации
    local distance = (position - currentPos).Magnitude
    
    print("📏 Расстояние до цели:", math.floor(distance), "стадов")
    
    -- Выбираем метод телепорта в зависимости от расстояния
    if distance < 10 then
        -- Очень близко - мгновенный телепорт с сохранением ориентации
        print("🚶 Очень близко, мгновенный телепорт")
        hrp.CFrame = targetCF
        task.wait(0.3)
    elseif distance < 30 then
        -- Близко - быстрый но безопасный телепорт
        print("🏃 Близко, быстрый телепорт (1с)")
        local success = pcall(function()
            smoothTweenTeleport(currentCF, targetCF, 1.0, Enum.EasingStyle.Linear)
        end)
        if not success then
            hrp.CFrame = targetCF
        end
        task.wait(0.4)
    else
        -- ПРОДВИНУТЫЙ ОБХОД - используем новую функцию с имитацией человека
        print("🛡️ ПРОДВИНУТЫЙ ОБХОД АНТИЧИТА")
        local success = pcall(function()
            advancedSafeTeleport(currentCF, targetCF, distance)
        end)
        if not success then
            -- Если не сработало, пробуем обычный медленный телепорт
            print("⚠️ Запасной вариант")
            smoothTeleport(currentPos, position, 15, 0.12)
        end
        task.wait(0.5)
    end
    
    return true
end
    end
    
    return true
end

-- Взять предмет из слота (нажать E на манекене)
local function takeItem(slotId, slot)
    print("📥 Беру предмет из слота, SlotId:", slotId)
    
    -- Ищем ProximityPrompt в слоте
    local prompt = nil
    for _, desc in pairs(slot:GetDescendants()) do
        if desc:IsA("ProximityPrompt") and desc.Enabled then
            prompt = desc
            print("🔘 Найден ProximityPrompt:", desc.ObjectText or "нет текста")
            break
        end
    end
    
    if not prompt then
        print("⚠️ ProximityPrompt не найден в слоте!")
        return false
    end
    
    -- Активируем ProximityPrompt (симулируем МГНОВЕННОЕ нажатие E)
    print("⚨️ Мгновенно нажимаю E на предмете...")
    local success = pcall(function()
        if fireproximityprompt then
            fireproximityprompt(prompt, 0) -- 0 = мгновенно без задержки
            print("✅ Нажал E мгновенно через fireproximityprompt")
        else
            -- Мгновенная активация без удержания
            prompt:InputHoldBegin()
            prompt:InputHoldEnd()
            print("✅ Нажал E мгновенно")
        end
    end)
    
    if not success then
        print("❌ Ошибка активации ProximityPrompt")
        return false
    end
    
    -- СИЛЬНО увеличенная задержка для корзины (чтобы вещь ТОЧНО попала)
    task.wait(1.8)
    print("✅ Предмет должен быть в корзине")
    
    return true
end

-- Оплатить предмет (нажать PAY в GUI)
local function purchaseItem(slotId)
    print("💳 Оплачиваю предмет, SlotId:", slotId)
    
    local playerGui = player:WaitForChild("PlayerGui")
    
    -- СИЛЬНО увеличенная задержка для загрузки GUI
    task.wait(0.8)
    
    -- Ищем кнопку PAY
    print("🔍 Ищу кнопку PAY в GUI...")
    local payButtonFound = false
    
    for _, gui in pairs(playerGui:GetChildren()) do
        if gui:IsA("ScreenGui") and gui.Enabled then
            for _, button in pairs(gui:GetDescendants()) do
                if button:IsA("TextButton") and button.Visible then
                    local text = button.Text:upper()
                    local name = button.Name:upper()
                    
                    if text:find("PAY") or name:find("PAY") then
                        print("💳 Найдена кнопка PAY:", button.Name, "Текст:", button.Text)
                        payButtonFound = true
                        
                        -- Нажимаем кнопку через getconnections
                        pcall(function()
                            if getconnections then
                                for _, connection in pairs(getconnections(button.MouseButton1Click)) do
                                    connection:Fire()
                                    print("✅ Нажал PAY через getconnections")
                                end
                            end
                        end)
                        
                        -- Также пробуем обычный клик
                        pcall(function()
                            button.MouseButton1Click:Fire()
                        end)
                        
                        task.wait(0.7) -- СИЛЬНО увеличенная задержка после нажатия PAY
                        break
                    end
                end
            end
            if payButtonFound then break end
        end
    end
    
    -- Отправляем RemoteFunction для покупки
    task.wait(0.8) -- СИЛЬНО увеличенная задержка перед RemoteFunction
    print("📡 Вызываю RemoteFunction ShopPurchase...")
    local success, result = pcall(function()
        local args = {"ShopPurchase", {slotId}}
        return game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("RemoteFunction"):InvokeServer(unpack(args))
    end)
    
    if success then
        print("✅ Покупка выполнена! Результат:", result)
        return true
    else
        print("❌ Ошибка покупки:", result)
        return false
    end
end

-- Купить один предмет
local function buyItem(mannequinData)
    if autoBuyRunning then
        return false, "Автобай уже работает"
    end
    
    autoBuyRunning = true
    
    local slot = mannequinData.slot
    local mannequin = mannequinData.mannequin
    local clothingData = mannequinData.clothingData
    
    -- ==========================================
    -- ПРОВЕРКА СУЩЕСТВОВАНИЯ ПЕРЕД ПОКУПКОЙ
    -- ==========================================
    
    -- ПРОВЕРКА 1: Манекен все еще существует?
    local mannequinExists = pcall(function()
        return mannequin and mannequin.Parent and mannequin:IsDescendantOf(workspace)
    end)
    
    if not mannequinExists or not mannequin or not mannequin.Parent then
        autoBuyRunning = false
        print("⚠️ Манекен уже не существует (вероятно уже куплен)")
        forceRemoveESP(mannequin, "манекен исчез перед покупкой")
        return false, "Манекен не существует"
    end
    
    -- ПРОВЕРКА 2: Слот все еще существует?
    local slotExists = pcall(function()
        return slot and slot.Parent and slot:IsDescendantOf(workspace)
    end)
    
    if not slotExists or not slot or not slot.Parent then
        autoBuyRunning = false
        print("⚠️ Слот не существует (вешалка пропала)")
        forceRemoveESP(mannequin, "слот исчез перед покупкой")
        return false, "Слот не существует"
    end
    
    -- ПРОВЕРКА 3: Манекен все еще в слоте?
    local mannequinInSlot = slot:FindFirstChild("Mannequin")
    if not mannequinInSlot or mannequinInSlot ~= mannequin then
        autoBuyRunning = false
        print("⚠️ Манекен не в слоте (уже куплен другим игроком)")
        forceRemoveESP(mannequin, "манекен не в слоте")
        return false, "Манекен не в слоте"
    end
    
    -- ПРОВЕРКА 4: Одежда все еще на манекене?
    local hasTargetClothing = false
    for _, child in pairs(mannequin:GetChildren()) do
        if child:IsA("Shirt") then
            local id = child.ShirtTemplate:match("%d+")
            if id and ASSETS[id] then
                hasTargetClothing = true
                break
            end
        elseif child:IsA("Pants") then
            local id = child.PantsTemplate:match("%d+")
            if id and ASSETS[id] then
                hasTargetClothing = true
                break
            end
        end
    end
    
    if not hasTargetClothing then
        autoBuyRunning = false
        print("⚠️ Одежда уже не на манекене (куплена)")
        forceRemoveESP(mannequin, "одежда исчезла")
        return false, "Одежда не найдена"
    end
    
    -- ==========================================
    -- ВСЕ ПРОВЕРКИ ПРОЙДЕНЫ - НАЧИНАЕМ ПОКУПКУ
    -- ==========================================
    
    -- Получаем имя вещи
    local itemNames = {}
    if clothingData.shirtId then
        local item = ASSETS[clothingData.shirtId]
        if item then table.insert(itemNames, item.name) end
    end
    if clothingData.pantsId then
        local item = ASSETS[clothingData.pantsId]
        if item then table.insert(itemNames, item.name) end
    end
    local itemName = table.concat(itemNames, " + ")
    
    -- Проверяем баланс и цену
    local playerMoney = getPlayerMoney()
    local itemPrice = getItemPrice(slot)
    
    print("")
    print("═══════════════════════════════")
    print("🛒 ПОКУПАЮ:", itemName)
    print("💰 Баланс:", playerMoney, "$")
    print("💵 Цена:", itemPrice, "$")
    
    -- Проверка хватает ли денег
    if itemPrice > 0 and playerMoney < itemPrice then
        autoBuyRunning = false
        print("❌ НЕ ХВАТАЕТ ДЕНЕГ!")
        print("   Нужно:", itemPrice, "$")
        print("   Есть:", playerMoney, "$")
        print("   Не хватает:", itemPrice - playerMoney, "$")
        print("═══════════════════════════════")
        return false, "Не хватает денег"
    end
    
    if itemPrice > 0 then
        print("✅ Денег хватает!")
    else
        print("⚠️ Цена не найдена, пробую купить")
    end
    print("═══════════════════════════════")
    
    -- Получаем SlotId
    local slotId = getSlotId(slot)
    if not slotId then
        autoBuyRunning = false
        print("❌ SlotId не найден")
        return false, "SlotId не найден"
    end
    print("🆔 SlotId:", slotId)
    
    -- Получаем зону
    local zoneName = slot.Parent.Parent.Name
    print("📍 Зона:", zoneName)
    
    -- Шаг 1: Телепорт к предмету
    print("")
    print("🚀 ШАГ 1: Телепортируюсь к предмету...")
    local mannequinPos = mannequin:FindFirstChild("HumanoidRootPart") or mannequin:FindFirstChild("Torso") or mannequin.PrimaryPart
    if not mannequinPos then
        autoBuyRunning = false
        print("❌ Позиция манекена не найдена")
        return false, "Позиция манекена не найдена"
    end
    
    -- Телепортируемся прямо к манекену (еще ближе)
    local targetPos = mannequinPos.Position + Vector3.new(0, 0, 1.5)
    if not teleportTo(targetPos) then
        autoBuyRunning = false
        print("❌ Ошибка телепорта к предмету")
        return false, "Ошибка телепорта к предмету"
    end
    print("✅ Телепортирован к предмету")
    
    task.wait(1.2) -- СИЛЬНО увеличенная задержка после телепорта к предмету
    
    -- Шаг 2: Взять предмет (нажать E)
    print("")
    print("👕 ШАГ 2: Нажимаю E на предмете...")
    if not takeItem(slotId, slot) then
        autoBuyRunning = false
        print("❌ Не удалось взять предмет")
        return false, "Не удалось взять предмет"
    end
    print("✅ Предмет взят и в корзине")
    
    task.wait(1.5) -- СИЛЬНО увеличенная задержка после взятия предмета
    
    -- Шаг 3: Найти NPC
    print("")
    print("🔍 ШАГ 3: Ищу NPC в зоне...")
    local npc = findNPCInZone(zoneName)
    if not npc then
        autoBuyRunning = false
        print("❌ NPC не найден в зоне")
        return false, "NPC не найден в зоне"
    end
    print("✅ NPC найден")
    
    -- Шаг 4: Телепорт к NPC
    print("")
    print("🚀 ШАГ 4: Телепортируюсь к NPC...")
    local npcTargetPos = npc.Position + Vector3.new(0, 0, 3)
    if not teleportTo(npcTargetPos) then
        autoBuyRunning = false
        print("❌ Ошибка телепорта к NPC")
        return false, "Ошибка телепорта к NPC"
    end
    print("✅ Телепортирован к NPC")
    
    task.wait(1.0) -- СИЛЬНО увеличенная задержка после телепорта к NPC
    
    -- Шаг 5: Открыть магазин (нажать E на NPC)
    print("")
    print("� ШАГ 5: Нажимаю E на NPC...")
    if not openShopNPC(npc) then
        autoBuyRunning = false
        print("❌ Не удалось открыть магазин")
        return false, "Не удалось открыть магазин"
    end
    print("✅ Магазин открыт")
    
    task.wait(1.2) -- СИЛЬНО увеличенная задержка после открытия магазина
    
    -- Шаг 6: Оплата (нажать PAY)
    print("")
    print("💳 ШАГ 6: Нажимаю PAY...")
    if not purchaseItem(slotId) then
        autoBuyRunning = false
        print("❌ Ошибка при оплате")
        return false, "Ошибка при оплате"
    end
    
    task.wait(1.0) -- СИЛЬНО увеличенная задержка после оплаты
    
    -- ДОБАВЛЯЕМ СЛОТ В СПИСОК КУПЛЕННЫХ
    purchasedSlots[slotId] = true
    print("📋 Слот", slotId, "добавлен в список купленных")
    
    -- УДАЛЯЕМ ESP после успешной покупки
    forceRemoveESP(mannequin, "успешно куплено")
    
    autoBuyRunning = false
    print("")
    print("═══════════════════════════════")
    print("✅ УСПЕШНО КУПЛЕНО:", itemName)
    print("═══════════════════════════════")
    print("")
    return true, "Успешно куплено"
end

-- Автобай всех найденных вещей
local function startAutoBuy()
    if not espEnabled then
        return false, "Сначала включи ESP!"
    end
    
    autoBuyEnabled = true
    print("🤖 Автобай запущен!")
    print("💰 Проверяю баланс...")
    
    local playerMoney = getPlayerMoney()
    print("💵 Ваш баланс:", playerMoney, "$")
    
    task.spawn(function()
        local mannequins = getAllMannequins()
        local bought = 0
        local errors = 0
        local skippedNoMoney = 0
        
        -- Сортируем по цене (от дешевых к дорогим)
        local sortedMannequins = {}
        for _, data in pairs(mannequins) do
            -- Проверяем что вещь в базе
            if (data.clothingData.shirtId and ASSETS[data.clothingData.shirtId]) or
               (data.clothingData.pantsId and ASSETS[data.clothingData.pantsId]) then
                
                local price = getItemPrice(data.slot)
                table.insert(sortedMannequins, {
                    data = data,
                    price = price
                })
            end
        end
        
        -- Сортируем по цене
        table.sort(sortedMannequins, function(a, b)
            return a.price < b.price
        end)
        
        print("🔍 Найдено вещей для покупки:", #sortedMannequins)
        print("")
        
        for _, item in pairs(sortedMannequins) do
            if not autoBuyEnabled then
                print("⏸️ Автобай остановлен")
                break
            end
            
            -- ============================================
            -- ПРОВЕРКА: УЖЕ КУПИЛИ ЭТОТ СЛОТ?
            -- ============================================
            local slotId = getSlotId(item.data.slot)
            if slotId and purchasedSlots[slotId] then
                -- Получаем имя вещи
                local itemNames = {}
                if item.data.clothingData.shirtId then
                    local asset = ASSETS[item.data.clothingData.shirtId]
                    if asset then table.insert(itemNames, asset.name) end
                end
                if item.data.clothingData.pantsId then
                    local asset = ASSETS[item.data.clothingData.pantsId]
                    if asset then table.insert(itemNames, asset.name) end
                end
                local itemName = table.concat(itemNames, " + ")
                
                print("⏭️ ПРОПУСК (уже куплено):", itemName)
                print("   SlotId:", slotId)
                forceRemoveESP(item.data.mannequin, "уже куплено ранее")
                task.wait(0.1)
                continue
            end
            
            -- ============================================
            -- ПРОВЕРКА СУЩЕСТВОВАНИЯ ПЕРЕД ПОПЫТКОЙ КУПИТЬ
            -- ============================================
            local mannequin = item.data.mannequin
            local slot = item.data.slot
            
            -- Проверяем что вещь все еще существует
            local itemStillExists = false
            pcall(function()
                if mannequin and mannequin.Parent and mannequin:IsDescendantOf(workspace) then
                    if slot and slot.Parent and slot:IsDescendantOf(workspace) then
                        local mannequinInSlot = slot:FindFirstChild("Mannequin")
                        if mannequinInSlot == mannequin then
                            -- Проверяем что одежда все еще на манекене
                            for _, child in pairs(mannequin:GetChildren()) do
                                if child:IsA("Shirt") then
                                    local id = child.ShirtTemplate:match("%d+")
                                    if id and ASSETS[id] then
                                        itemStillExists = true
                                        break
                                    end
                                elseif child:IsA("Pants") then
                                    local id = child.PantsTemplate:match("%d+")
                                    if id and ASSETS[id] then
                                        itemStillExists = true
                                        break
                                    end
                                end
                            end
                        end
                    end
                end
            end)
            
            if not itemStillExists then
                -- Вещь уже куплена или пропала - пропускаем и удаляем ESP
                print("⏭️ ПРОПУСК (вещь уже куплена или пропала)")
                forceRemoveESP(mannequin, "вещь пропала в автобае")
                updateESP() -- Принудительное обновление
                task.wait(0.1)
                continue
            end
            
            -- ============================================
            -- ПРОВЕРКА ДЕНЕГ
            -- ============================================
            
            -- Обновляем баланс перед каждой покупкой
            playerMoney = getPlayerMoney()
            
            -- Проверяем хватает ли денег
            if item.price > 0 and playerMoney < item.price then
                skippedNoMoney = skippedNoMoney + 1
                
                -- Получаем имя вещи
                local itemNames = {}
                if item.data.clothingData.shirtId then
                    local asset = ASSETS[item.data.clothingData.shirtId]
                    if asset then table.insert(itemNames, asset.name) end
                end
                if item.data.clothingData.pantsId then
                    local asset = ASSETS[item.data.clothingData.pantsId]
                    if asset then table.insert(itemNames, asset.name) end
                end
                local itemName = table.concat(itemNames, " + ")
                
                print("💸 ПРОПУСК (не хватает денег):", itemName)
                print("   Цена:", item.price, "$ | Баланс:", playerMoney, "$")
                print("")
                
                -- Если больше не можем купить ничего, останавливаемся
                if item.price <= playerMoney then
                    -- Есть деньги, продолжаем
                else
                    -- Если отсортировано правильно, все следующие будут еще дороже
                    print("⚠️ На остальные вещи тоже не хватит денег")
                    break
                end
            else
                local success, message = buyItem(item.data)
                if success then
                    bought = bought + 1
                    updateESP() -- Обновляем ESP после покупки
                else
                    if message == "Манекен не существует" or message == "Слот не существует" or 
                       message == "Манекен не в слоте" or message == "Одежда не найдена" then
                        -- Вещь уже куплена, не считаем ошибкой
                        print("⏭️ Вещь уже куплена")
                    elseif message ~= "Не хватает денег" then
                        errors = errors + 1
                        warn("❌", message)
                    end
                end
                
                task.wait(1.2) -- СИЛЬНО увеличенная задержка между покупками для безопасности
            end
        end
        
        autoBuyEnabled = false
        print("")
        print("🏁 Автобай завершен!")
        print("✅ Куплено:", bought)
        print("💸 Пропущено (нет денег):", skippedNoMoney)
        print("❌ Ошибок:", errors)
        print("💰 Остаток:", getPlayerMoney(), "$")
    end)
    
    return true, "Автобай запущен"
end

local function stopAutoBuy()
    autoBuyEnabled = false
    print("⏹️ Автобай остановлен")
end

-- Очистить список купленных предметов
local function clearPurchasedList()
    local count = 0
    for _ in pairs(purchasedSlots) do
        count = count + 1
    end
    purchasedSlots = {}
    print("🗑️ Список купленных очищен! Было:", count)
    return count
end

    if input.UserInputType == Enum.UserInputType.MouseButton1 then

        dragStart = input.Position
        startPos = frame.Position
    end

    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

frame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
