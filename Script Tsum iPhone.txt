-- ============================================
-- ПОЛУЧЕНИЕ HWID
-- ============================================
local function getHWID()
    local success, hwid = pcall(function()
        return game:GetService("RbxAnalyticsService"):GetClientId()
    end)
    return success and hwid or "unknown-hwid"
end

-- ============================================
-- HTTP REQUEST (ОБЁРТКА)
-- ============================================
local function makeRequest(endpoint, data)
    local success, result = pcall(function()
        local response = request({
            Url = SERVER_URL .. endpoint,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json"
            },
            Body = HttpService:JSONEncode(data)
        })
        return HttpService:JSONDecode(response.Body)
    end)
    
    if not success then
        warn("[ESP] Ошибка запроса:", result)
        return nil
    end
    
    return result
end

local function validateKey()
    print("[AUTH] Проверка ключа...")
    
    local authData = {
        key = USER_KEY,
        hwid = getHWID(),
        userId = player.UserId,
        game = game.PlaceId
    }
    
    local response = makeRequest("/api/validate", authData)
    
    if not response then
        player:Kick("❌ Не удалось подключиться к серверу")
        return false
    end
    
    if not response.success then
        -- Проверяем истёк ли ключ
        if response.expired or response.kickPlayer then
            -- Ключ истёк - кикаем с особым сообщением
            local kickMessage = response.message or "⏰ Срок действия ключа истёк"
            kickMessage = kickMessage .. "\n\n🔑 Купите новый ключ в loader'е"
            player:Kick(kickMessage)
        else
            -- Обычный кик
            player:Kick(response.message or "❌ Доступ запрещён")
        end
        return false
    end
    
    print("[AUTH] ✅ Авторизация успешна")
    print("[AUTH] Тип подписки:", response.tier)
    print("[AUTH] Истекает:", response.expiry)
    
    return true

    local function serverHop(findSmallest)
        if isServerHopping then
            warn("[SERVER HOP] Уже идёт поиск сервера...")
            return false, "Поиск уже запущен"
        end
        
        isServerHopping = true
        
        local success, result = pcall(function()
            local placeId = game.PlaceId
            local jobId = game.JobId
            
            print("[SERVER HOP] Получение списка серверов...")
            
            local sortOrder = findSmallest and "Asc" or "Desc"
            local httpData = getServersData(placeId, sortOrder)
            local serversData = HttpService:JSONDecode(httpData)
            
            if not serversData or not serversData.data then
                error("Некорректный ответ от API")
            end
            
            local servers = serversData.data
            
            if #servers == 0 then
                error("Нет доступных серверов")
            end
            
            -- Фильтруем сервера
            local validServers = {}
            for _, server in ipairs(servers) do
                -- Исключаем текущий сервер и полные сервера
                if server.id ~= jobId and server.playing < server.maxPlayers then
                    table.insert(validServers, {
                        id = server.id,
                        playing = server.playing,
                        maxPlayers = server.maxPlayers
                    })
                end
            end
            
            if #validServers == 0 then
                error("Нет подходящих серверов для телепорта")
            end
            
            -- Сортируем по количеству игроков
            table.sort(validServers, function(a, b)
                if findSmallest then
                    return a.playing < b.playing
                else
                    return a.playing > b.playing
                end
            end)
            
            -- Выбираем лучший сервер
            local targetServer = validServers[1]
            
            print(string.format("[SERVER HOP] Найден сервер: %d/%d игроков", 
                targetServer.playing, targetServer.maxPlayers))
            
            -- Телепортируем
            TeleportService:TeleportToPlaceInstance(
                placeId,
                targetServer.id,
                player
            )
            
            return targetServer
        end)
        
        isServerHopping = false
        
        if not success then
            warn("[SERVER HOP] Ошибка:", result)
            return false, result
        end
        
        return true, result
    end

    local function createAccessoryESP(data)
        local model = data.model
        local meshId = data.meshId
        local info = data.data
        
        -- Проверка что модель существует
        if not model or not model.Parent then return end

        local rarity = info.rarity or "Legendary"
        local chance = info.chance
        local color = RARITY_COLORS[rarity] or RARITY_COLORS.Legendary
        
        -- Проверяем что не создали уже ESP для этой модели
        if model:FindFirstChild("ESP_Highlight") then return end
        
        updateOrCreateHighlight(model)
        
        local chanceText = chance and (" | " .. formatChance(chance)) or ""
        updateOrCreateBillboard(model, name, chanceText, rarity, color, chance)

        if not foundItems[itemKey] then
            foundItems[itemKey] = {
                name = name,
                chance = chance,
                rarity = rarity,
                color = color,
                model = model
            }
        end
    end

    local function saveConfig()
        if not fileSystemSupported then
            warn("[CONFIG] Файловая система не поддерживается экзекутором")
            return false
        end
        
        local config = {
            -- Основные настройки
            espItemsEnabled = espItemsEnabled,
            espAccessoriesEnabled = espAccessoriesEnabled,
            chams = settings.chams,
            tracers = settings.tracers,
            highlightMode = settings.highlightMode,
            accessoriesHighlightMode = settings.accessoriesHighlightMode,
            
            -- Speed hack
            speedEnabled = settings.speedEnabled,
            walkSpeed = settings.walkSpeed,
            
            -- Fast Take
            fastTakeEnabled = settings.fastTakeEnabled,
            
            -- Custom Nickname
            customNickEnabled = customNickEnabled,
            customNickname = customNickname,
            tempNickname = tempNickname,
            
            -- Выбранные предметы
            selectedAssets = {},
            selectedAccessories = {}
        }
        
        -- Сохраняем выбранные предметы
        for id in pairs(selectedAssets) do
            table.insert(config.selectedAssets, id)
        end
        
        -- Сохраняем выбранные аксессуары
        for id in pairs(selectedAccessories) do
            table.insert(config.selectedAccessories, id)
        end
        
        local success, encoded = pcall(function()
            return HttpService:JSONEncode(config)
        end)
        
        if success then
            local writeSuccess = pcall(function()
                writefile(configFileName, encoded)
            end)
            
            if writeSuccess then
                print("[CONFIG] Конфигурация сохранена")
                return true
            else
                warn("[CONFIG] Ошибка записи файла")
                return false
            end
        else
            warn("[CONFIG] Ошибка кодирования конфигурации:", encoded)
            return false
        end
    end
    
    local function loadConfig()
        if not fileSystemSupported then
            print("[CONFIG] Файловая система не поддерживается, используем настройки по умолчанию")
            return nil
        end
        
        local fileExists = pcall(function()
            return isfile(configFileName)
        end)
        
        if not fileExists or not isfile(configFileName) then
            print("[CONFIG] Файл конфигурации не найден, используем настройки по умолчанию")
            return nil
        end
        
        local success, content = pcall(function()
            return readfile(configFileName)
        end)
        
        if not success then
            warn("[CONFIG] Ошибка чтения файла конфигурации")
            return nil
        end
        
        local decoded
        success, decoded = pcall(function()
            return HttpService:JSONDecode(content)
        end)
        
        if not success then
            warn("[CONFIG] Ошибка декодирования конфигурации")
            return nil
        end
        
        print("[CONFIG] ✅ Конфигурация загружена!")
        return decoded
    end

    local function tweenToObject(targetObj)
        local char = player.Character
        if not char then return false end
        local root = char:FindFirstChild("HumanoidRootPart")
        if not root then return false end
        
        local targetPart = targetObj
        if targetObj:IsA("Model") then
            targetPart = targetObj.PrimaryPart or targetObj:FindFirstChildWhichIsA("BasePart")
        end
        
        if not targetPart or not targetPart:IsA("BasePart") then
            print("[AutoBuy] Не удалось найти позицию для полета")
            return false
        end
        
        -- Вычисляем расстояние и время полета
        local distance = (root.Position - targetPart.Position).Magnitude
        local duration = distance / autoBuySpeed
        
        print(string.format("[AutoBuy] Лечу к цели (%.0f м, %.1f сек)", distance, duration))
        
        -- Создаем Tween для плавного перемещения
        local targetCFrame = CFrame.new(targetPart.Position) * CFrame.new(0, 3, 0)
        local tweenInfo = TweenInfo.new(duration, Enum.EasingStyle.Linear)
        local tween = TweenService:Create(root, tweenInfo, {CFrame = targetCFrame})
        
        tween:Play()
        tween.Completed:Wait()
        
        print("[AutoBuy] ✅ Прилетел к цели")
        return true
    end

    local function findShopZone(obj)
        local objPos = obj:IsA("Model") and (obj.PrimaryPart and obj.PrimaryPart.Position or (obj:FindFirstChildWhichIsA("BasePart") and obj:FindFirstChildWhichIsA("BasePart").Position)) or (obj:IsA("BasePart") and obj.Position)
        if not objPos then return nil end
        for _, zone in pairs(workspace:GetDescendants()) do
            if zone.Name:match("^Shop_ShopZone") and zone:IsA("BasePart") then
                local zonePos, zoneSize = zone.Position, zone.Size
                if math.abs(objPos.X - zonePos.X) <= zoneSize.X / 2 and math.abs(objPos.Y - zonePos.Y) <= zoneSize.Y / 2 and math.abs(objPos.Z - zonePos.Z) <= zoneSize.Z / 2 then
                    return zone
                end
            end
        end
        return nil
    end

    local function autoBuyCycle()
        if not autoBuyEnabled then return end
        
        print("\n[AutoBuy] ═══════════════════════════════")
        print("[AutoBuy] 🔄 Цикл покупки #" .. (autoBuyCounter + 1))
        
        local items = findAllItems()
        if #items == 0 then
            print("[AutoBuy] ⏸️ Нет предметов с ESP, жду...")
            return
        end
        
        print("[AutoBuy] 📊 Найдено предметов: " .. #items)
        
        table.sort(items, function(a, b)
            local distA = getDistance(a.object) or math.huge
            local distB = getDistance(b.object) or math.huge
            return distA < distB
        end)
        
        local targetItem = items[1]
        local distance = getDistance(targetItem.object)
        print(string.format("[AutoBuy] 🎯 Цель: %s (%s) - %.0f м", targetItem.name, targetItem.rarity, distance or 0))
        
        print("[AutoBuy] ✈️ Лечу к предмету...")
        local success = tweenToObject(targetItem.object)
        if not success then
            print("[AutoBuy] ❌ Не удалось долететь до предмета")
            return
        end
        
        task.wait(0.5)
        
        print("[AutoBuy] 🛒 Беру предмет в корзину...")
        if not clickObject(targetItem.object) then
            print("[AutoBuy] ❌ Не удалось взять предмет")
            return
        end
        
        print("[AutoBuy] ✅ Предмет взят в корзину")
        task.wait(0.5)
        
        local excludedSeller = nil
        local maxRetries = 3
        local retryCount = 0
        
        while retryCount < maxRetries do
            print("[AutoBuy] 🔍 Ищу продавца (Seller)...")
            if excludedSeller then
                print("[AutoBuy] ⚠️ Исключаю неправильного Seller из поиска")
            end
            
            local seller = findSellerInShop(targetItem.object, excludedSeller)
            if not seller then
                print("[AutoBuy] ❌ Seller не найден в магазине")
                return
            end
            
            print("[AutoBuy] ✈️ Лечу к продавцу...")
            local successFly = tweenToObject(seller)
            if not successFly then
                print("[AutoBuy] ❌ Не удалось долететь до Seller")
                return
            end
            
            task.wait(0.3)
            
            print("[AutoBuy] 💰 Покупаю предмет...")
            if not clickObject(seller) then
                print("[AutoBuy] ❌ Не удалось кликнуть по Seller")
                return
            end
            
            task.wait(1.5)
            
            local itemsAfter = findAllItems()
            local itemStillExists = false
            
            for _, item in pairs(itemsAfter) do
                if item.assetId == targetItem.assetId then
                    local dist = getDistance(item.object)
                    if dist and dist < 100 then
                        itemStillExists = true
                        targetItem = item
                        break
                    end
                end
            end
            
            if not itemStillExists then
                autoBuyCounter = autoBuyCounter + 1
                
                -- Сохраняем в историю
                local timeStr = os.date("%H:%M:%S")
                table.insert(autoBuyHistory, {
                    name = targetItem.name,
                    rarity = targetItem.rarity,
                    time = timeStr,
                    assetId = targetItem.assetId
                })
                
                -- Обновляем GUI
                updateAutoBuyStatsGUI()
                
                print("[AutoBuy] ✅✅✅ ПОКУПКА ЗАВЕРШЕНА! ✅✅✅")
                print("[AutoBuy] 📈 Всего куплено: " .. autoBuyCounter)
                print("[AutoBuy] ═══════════════════════════════\n")
                break
            else
                print("[AutoBuy] ⚠️⚠️⚠️ КОРЗИНА ПУСТАЯ! Предмет не был куплен!")
                print("[AutoBuy] 🔄 Возвращаюсь к предмету и ищу другого Seller...")
                
                excludedSeller = seller
                retryCount = retryCount + 1
                
                if retryCount >= maxRetries then
                    print("[AutoBuy] ❌ Превышено количество попыток, пропускаю предмет")
                    return
                end
                
                print("[AutoBuy] ✈️ Лечу обратно к предмету...")
                local successReturn = tweenToObject(targetItem.object)
                if not successReturn then
                    print("[AutoBuy] ❌ Не удалось вернуться к предмету")
                    return
                end
                
                task.wait(0.5)
                
                print("[AutoBuy] 🛒 Беру предмет в корзину заново...")
                if not clickObject(targetItem.object) then
                    print("[AutoBuy] ❌ Не удалось взять предмет повторно")
                    return
                end
                
                print("[AutoBuy] ✅ Предмет снова в корзине, ищу другого Seller...")
                task.wait(0.5)
            end
        end
        
        task.wait(0.5)
        if espEnabled then
            updateESP()
        end
    end

    local function startAutoBuy()
        if autoBuyThread then return end
        
        print("[AutoBuy] 🔍 Проверка доступных функций:")
        print("  fireclickdetector:", fireclickdetector ~= nil)
        print("  fireproximityprompt:", fireproximityprompt ~= nil)
        print("  firetouchinterest:", firetouchinterest ~= nil)
        print("  mouse1click:", mouse1click ~= nil)
        print("  mouse1press:", mouse1press ~= nil)
        print("  mouse1release:", mouse1release ~= nil)
        
        print("[AutoBuy] 🚀 AutoBuy запущен")
        autoBuyEnabled = true
        
        -- Создаем GUI статистики
        createAutoBuyStatsGUI()
        updateAutoBuyStatsGUI()
        
        autoBuyThread = task.spawn(function()
            while autoBuyEnabled do
                local success, err = pcall(function()
                    autoBuyCycle()
                end)
                
                if not success then
                    print("[AutoBuy] ⚠️ Ошибка в цикле: " .. tostring(err))
                end
                
                task.wait(1)
            end
            autoBuyThread = nil
            print("[AutoBuy] ⏹️ AutoBuy остановлен")
        end)
    end

    local function stopAutoBuy()

        if autoBuyThread then

        end
        
        local camera = workspace.CurrentCamera
        camera.CameraType = Enum.CameraType.Custom

        -- Удаляем GUI статистики
        if autoBuyStatsGUI then
            autoBuyStatsGUI:Destroy()
            autoBuyStatsGUI = nil
        end
    end

    -- Функция обновления информации о ключе
    local function updateKeyInfo()
        local response = makeRequest("/api/key-info", { key = USER_KEY })
        
        if not response then
            keyInfoLabel:Set("❌ Не удалось получить информацию о ключе")
            keyTypeLabel:Set("Тип ключа: -")
            keyStatusLabel:Set("Статус: -")
            timeLeftLabel:Set("Осталось времени: -")
            devicesLabel:Set("Устройств: -")
            daysLabel:Set("Дней: -")
            hoursLabel:Set("Часов: -")
            minutesLabel:Set("Минут: -")
            activatedLabel:Set("Активирован: -")
            tierLabel:Set("Уровень: -")
            return
        end
        
        if not response.success then
            keyInfoLabel:Set("❌ " .. (response.message or "Ошибка"))
            keyTypeLabel:Set("Тип ключа: -")
            keyStatusLabel:Set("Статус: -")
            timeLeftLabel:Set("Осталось времени: -")
            devicesLabel:Set("Устройств: -")
            daysLabel:Set("Дней: -")
            hoursLabel:Set("Часов: -")
            minutesLabel:Set("Минут: -")
            activatedLabel:Set("Активирован: -")
            tierLabel:Set("Уровень: -")
            return
        end
        
        -- Обновляем основную информацию
        keyInfoLabel:Set("✅ Ключ действителен")
        keyTypeLabel:Set("Тип ключа: " .. (response.type or "Неизвестно"))
        
        if response.isPermanent then
            keyStatusLabel:Set("Статус: ✅ Активен (бессрочно)")
            timeLeftLabel:Set("Осталось времени: ∞ (Бессрочный)")
            daysLabel:Set("Дней: ∞")
            hoursLabel:Set("Часов: ∞")
            minutesLabel:Set("Минут: ∞")
        elseif response.timeLeft then
            local tl = response.timeLeft
            keyStatusLabel:Set("Статус: ✅ Активен")
            timeLeftLabel:Set("Осталось времени: " .. (tl.display or "0ч 0м"))
            daysLabel:Set("Дней: " .. tostring(tl.days or 0))
            hoursLabel:Set("Часов: " .. tostring(tl.hours or 0))
            hoursLabel:Set("Часов: " .. tostring(tl.hours or 0) .. " (всего " .. tostring(tl.totalHours or 0) .. ")")
            minutesLabel:Set("Минут: " .. tostring(tl.minutes or 0) .. " (всего " .. tostring(tl.totalMinutes or 0) .. ")")
        else
            keyStatusLabel:Set("Статус: ⏳ Не активирован")
            timeLeftLabel:Set("Осталось времени: Ключ не активирован")
            daysLabel:Set("Дней: -")
            hoursLabel:Set("Часов: -")
            minutesLabel:Set("Минут: -")
        end
        
        devicesLabel:Set("Устройств: " .. tostring(response.devices or 0) .. " / " .. tostring(response.maxDevices or 1))
        
        -- Дополнительная информация
        if response.activatedAt then
            local activatedDate = os.date("%d.%m.%Y %H:%M:%S", response.activatedAt / 1000)
            activatedLabel:Set("Активирован: " .. activatedDate)
        else
            activatedLabel:Set("Активирован: Не активирован")
        end
        
        tierLabel:Set("Уровень: " .. string.upper(response.tier or "PREMIUM"))
    end
