-- ============================================
-- ФУНКЦИИ ПОИСКА
-- ============================================
local function getAllMannequins()
    local results = {}
    for _, zone in pairs(workspace:GetChildren()) do
        if zone:IsA("Model") and zone.Name:match("Shop_ShopZone") then
            local slots = zone:FindFirstChild("ItemSlots")
            if slots then
                for _, slot in pairs(slots:GetChildren()) do
                    if slot.Name:match("Slot_") then
                        local mannequin = slot:FindFirstChild("Mannequin")
                        if mannequin and mannequin:IsA("Model") then
                            local clothingData = {}
                            for _, child in pairs(mannequin:GetChildren()) do
                                if child:IsA("Shirt") then
                                    clothingData.shirtId = child.ShirtTemplate:match("%d+")
                                elseif child:IsA("Pants") then
                                    clothingData.pantsId = child.PantsTemplate:match("%d+")
                                end
                            end
                            if (clothingData.shirtId and ASSETS[clothingData.shirtId]) or
                               (clothingData.pantsId and ASSETS[clothingData.pantsId]) then
                                table.insert(results, {
                                    mannequin = mannequin,
                                    clothingData = clothingData,
                                    slot = slot
                                })
                            end
                        end
                    end
                end
            end
        end
    end
    return results
end

local function getAllAccessories()
    local results = {}
    for _, zone in pairs(workspace:GetChildren()) do
        if zone:IsA("Model") and zone.Name:match("Shop_ShopZone") then
            for _, child in pairs(zone:GetChildren()) do
                if child:IsA("Model") and child.Name:match("^M_") then
                    local target = child:FindFirstChild("Handle") or child:FindFirstChild("Body")
                    if target and target:IsA("MeshPart") and target.MeshId then
                        local id = target.MeshId:match("%d+")
                        if id and ACCESSORIES[id] then
                            table.insert(results, {
                                model = child,
                                meshId = id
                            })
                        end
                    end
                end
            end
        end
    end
    return results
end

-- ============================================
-- ESP ФУНКЦИИ
-- ============================================
local function createHighlight(model)
    if model:FindFirstChild("ESP_H") then return end
    local h = Instance.new("Highlight")
    h.Name = "ESP_H"
    h.Adornee = model
    h.FillColor = RARITY_COLOR
    h.OutlineColor = RARITY_COLOR
    h.FillTransparency = 0.5
    h.OutlineTransparency = 0
    h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    h.Parent = model
    table.insert(espObjects, h)
end

local function createLabel(model, text)
    if model:FindFirstChild("ESP_L") then return end
    local gui = Instance.new("BillboardGui")
    gui.Name = "ESP_L"
    gui.Adornee = model
    gui.Size = UDim2.new(0, 200, 0, 40)
    gui.StudsOffset = Vector3.new(0, 3, 0)
    gui.AlwaysOnTop = true
    gui.Parent = player:WaitForChild("PlayerGui")
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    label.BackgroundTransparency = 0.3
    label.TextColor3 = Color3.fromRGB(255, 255, 255)
    label.Font = Enum.Font.SourceSansBold
    label.TextSize = 14
    label.Text = text
    label.Parent = gui
    
    table.insert(espObjects, gui)
end

local function clearESP()
    for _, obj in pairs(espObjects) do
        if obj and obj.Parent then
            obj:Destroy()
        end
    end
    espObjects = {}
    espTracking = {}
end

local function enableESP()
    if espEnabled then return end
    espEnabled = true
    clearESP()
    
    local found = 0
    
    -- Вещи
    for _, data in pairs(getAllMannequins()) do
        local names = {}
        if data.clothingData.shirtId then
            local item = ASSETS[data.clothingData.shirtId]
            if item then table.insert(names, item.name) end
        end
        if data.clothingData.pantsId then
            local item = ASSETS[data.clothingData.pantsId]
            if item then table.insert(names, item.name) end
        end
        
        if #names > 0 then
            createHighlight(data.mannequin)
            createLabel(data.mannequin, table.concat(names, " + "))
            found = found + 1
        end
    end
    
    -- Аксессуары
    for _, data in pairs(getAllAccessories()) do
        local item = ACCESSORIES[data.meshId]
        if item then
            createHighlight(data.model)
            createLabel(data.model, item.name)
            found = found + 1

local function getPlayerMoney()
    local money = 0
    pcall(function()
        local leaderstats = player:FindFirstChild("leaderstats")
        if leaderstats then
            local cash = leaderstats:FindFirstChild("Cash") or leaderstats:FindFirstChild("Money")
            if cash then money = cash.Value end
        end
    end)
    return money
end

local function getItemPrice(slot)
    local price = 0
    pcall(function()
        if slot:GetAttribute("Price") then
            price = slot:GetAttribute("Price")
        end
    end)
    return price
end

local function getSlotId(slot)
    if slot:GetAttribute("SlotId") then
        return slot:GetAttribute("SlotId")
    end
    if slot.Name:match("Slot_%d+_%d+") then
        return slot.Name
    end
    return slot.Name
end

local function teleportTo(position)
    local character = player.Character
    if not character then return false end
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    hrp.CFrame = CFrame.new(position)
    task.wait(0.15)
    return true
end

-- Найти NPC в зоне
local function findNPCInZone(zoneName)
    local zone = workspace:FindFirstChild(zoneName)
    if not zone then return nil end
    
    for _, child in pairs(zone:GetDescendants()) do
        if child:IsA("Model") and child.Name:lower():find("npc") then
            local humanoidRootPart = child:FindFirstChild("HumanoidRootPart")
            if humanoidRootPart then
                return humanoidRootPart
            end
        end
    end
    return nil
end

-- Взять предмет из слота (нажать E на манекене)
local function takeItem(slotId, slot)
    print("📥 Беру предмет из слота, SlotId:", slotId)
    
    local prompt = nil
    for _, desc in pairs(slot:GetDescendants()) do
        if desc:IsA("ProximityPrompt") and desc.Enabled then
            prompt = desc
            print("🔘 Найден ProximityPrompt")
            break
        end
    end
    
    if not prompt then
        print("⚠️ ProximityPrompt не найден в слоте!")
        return false
    end
    
    print("⚡ Мгновенно нажимаю E на предмете...")
    local success = pcall(function()
        if fireproximityprompt then
            fireproximityprompt(prompt, 0)
            print("✅ Нажал E через fireproximityprompt")
        else
            prompt:InputHoldBegin()
            prompt:InputHoldEnd()
            print("✅ Нажал E")
        end
    end)
    
    if not success then
        print("❌ Ошибка активации ProximityPrompt")
        return false
    end
    
    task.wait(0.3)
    print("✅ Предмет должен быть в корзине")
    return true
end

local function buyItem(mannequinData)
    if autoBuyRunning then
        return false, "Автобай уже работает"
    end
    
    autoBuyRunning = true
    
    local slot = mannequinData.slot
    local mannequin = mannequinData.mannequin
    local clothingData = mannequinData.clothingData
    
    -- Проверка существования манекена
    if not mannequin or not mannequin.Parent then
        autoBuyRunning = false
        print("⚠️ Манекен уже не существует")
        return false, "Манекен не существует"
    end
    
    -- Получаем имя вещи
    local itemNames = {}
    if clothingData.shirtId then
        local item = ASSETS[clothingData.shirtId]
        if item then table.insert(itemNames, item.name) end
    end
    if clothingData.pantsId then
        local item = ASSETS[clothingData.pantsId]
        if item then table.insert(itemNames, item.name) end
    end
    local itemName = table.concat(itemNames, " + ")
    
    -- Проверка баланса
    local playerMoney = getPlayerMoney()
    local itemPrice = getItemPrice(slot)
    
    print("")
    print("═══════════════════════════════")
    print("🛒 ПОКУПАЮ:", itemName)
    print("💰 Баланс:", playerMoney, "$")
    print("💵 Цена:", itemPrice, "$")
    
    if itemPrice > 0 and playerMoney < itemPrice then
        autoBuyRunning = false
        print("❌ НЕ ХВАТАЕТ ДЕНЕГ!")
        print("═══════════════════════════════")
        return false, "Недостаточно денег"
    end
    
    print("✅ Денег достаточно!")
    print("═══════════════════════════════")
    
    -- Получаем SlotId
    local slotId = getSlotId(slot)
    if not slotId then
        autoBuyRunning = false
        print("❌ SlotId не найден")
        return false, "SlotId не найден"
    end
    print("🆔 SlotId:", slotId)
    
    -- Получаем зону
    local zoneName = slot.Parent.Parent.Name
    print("📍 Зона:", zoneName)
    
    -- Шаг 1: Телепорт к предмету
    print("")
    print("🚀 ШАГ 1: Телепортируюсь к предмету...")
    local mannequinPos = mannequin:FindFirstChild("HumanoidRootPart") or mannequin:FindFirstChild("Torso") or mannequin.PrimaryPart

        autoBuyRunning = false
        print("❌ Позиция манекена не найдена")
        return false, "Позиция манекена не найдена"
    end
    
    local targetPos = mannequinPos.Position + Vector3.new(0, 0, 1.5)

        autoBuyRunning = false
        print("❌ Ошибка телепорта к предмету")
        return false, "Ошибка телепорта к предмету"
    end
    print("✅ Телепортирован к предмету")
    
    task.wait(0.2)
    
    -- Шаг 2: Взять предмет (нажать E)
    print("")
    print("👕 ШАГ 2: Нажимаю E на предмете...")
    if not takeItem(slotId, slot) then
        autoBuyRunning = false
        print("❌ Не удалось взять предмет")
        return false, "Не удалось взять предмет"

    print("✅ Предмет взят и в корзине")

    if not teleportTo(targetPos) then
        autoBuyRunning = false
        print("❌ Ошибка телепорта к аксессуару")
        return false, "Ошибка телепорта к аксессуару"
    end
    print("✅ Телепортирован к аксессуару")
    
    task.wait(0.2)
    
    -- Шаг 2: Взять аксессуар (нажать E)
    print("")
    print("👜 ШАГ 2: Нажимаю E на аксессуаре...")
    local prompt = nil
    for _, desc in pairs(model:GetDescendants()) do
        if desc:IsA("ProximityPrompt") and desc.Enabled then
            prompt = desc
            print("🔘 Найден ProximityPrompt")
            break
        end
    end
    
    if not prompt then
        autoBuyRunning = false
        print("⚠️ ProximityPrompt не найден у аксессуара!")
        return false, "ProximityPrompt не найден"

        autoBuyRunning = false
        print("❌ Ошибка активации ProximityPrompt")

    print("✅ Аксессуар взят и в корзине")

    if not npc then

            end
        end

        autoBuyRunning = false
        print("❌ NPC не найден")
        return false, "NPC не найден"
    end
    print("✅ NPC найден")

    print("")
    print("🚀 ШАГ 4: Телепортируюсь к NPC...")

        print("❌ Ошибка телепорта к NPC")
        return false, "Ошибка телепорта к NPC"
    end

    print("")
    print("🛒 ШАГ 5: Нажимаю E на NPC...")
    if not openShopNPC(npc) then
        autoBuyRunning = false
        print("❌ Не удалось открыть магазин")
        return false, "Не удалось открыть магазин"
    end
    print("✅ Магазин открыт")
    
    task.wait(0.2)

    print("")

local function startAutoBuy()
    if not espEnabled then
        return false, "Сначала включи ESP!"
    end
    
    autoBuyEnabled = true
    print("🤖 Автобай запущен!")
    print("💰 Проверяю баланс...")
    
    local playerMoney = getPlayerMoney()
    print("💵 Ваш баланс:", playerMoney, "$")
    
    task.spawn(function()
        local mannequins = getAllMannequins()
        local accessories = getAllAccessories()
        local bought = 0
        local skipped = 0
        
        print("🔍 Найдено вещей:", #mannequins)
        print("🔍 Найдено аксессуаров:", #accessories)
        print("")
        
        -- Покупка вещей
        if #mannequins > 0 then
            print("👕 ========== ПОКУПАЮ ВЕЩИ ==========")
            for _, data in pairs(mannequins) do
                if not autoBuyEnabled then
                    print("⏸️ Автобай остановлен")
                    break
                end
                
                local success, message = buyItem(data)
                if success then
                    bought = bought + 1
                elseif message == "Недостаточно денег" then
                    skipped = skipped + 1
                end
                
                task.wait(0.5)
            end
        end
        
        -- Покупка аксессуаров
        if autoBuyEnabled and #accessories > 0 then
            print("")
            print("👜 ========== ПОКУПАЮ АКСЕССУАРЫ ==========")
            for _, data in pairs(accessories) do
                if not autoBuyEnabled then
                    print("⏸️ Автобай остановлен")
                    break
                end
                
                local success, message = buyAccessory(data)
                if success then
                    bought = bought + 1
                elseif message == "Недостаточно денег" then
                    skipped = skipped + 1
                end
                
                task.wait(0.5)
            end
        end
        
        autoBuyEnabled = false
        print("")
        print("🏁 Автобай завершен!")
        print("✅ Куплено:", bought)
        print("⏭️ Пропущено (нет денег):", skipped)
    end)
    
    return true, "Автобай запущен"
end

local function stopAutoBuy()
    autoBuyEnabled = false
    autoBuyRunning = false
    print("⏹️ Автобай остановлен")
end
